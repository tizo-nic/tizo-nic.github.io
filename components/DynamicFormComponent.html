import {
  Component,
  EventEmitter,
  Input,
  OnChanges,
  OnInit,
  Output,
  SimpleChanges
} from '@angular/core';
import {CountryISO} from "ngx-intl-tel-input";
import {UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from "@angular/forms";
import {Observable, map} from "rxjs";

import {FormInterface} from "../../schemes/interface/form.interface";
import {FormSelectable} from "../../schemes/support/form-selectable";
import {FormSegment} from "../../schemes/support/form-segment";

import * as moment from 'moment';

@Component({
  selector: 'app-dynamic-form',
  templateUrl: './dynamic-form.component.html',
  styleUrls: ['./dynamic-form.component.scss']
})
export class DynamicFormComponent implements OnInit, OnChanges{

  public segments: FormSegment[] = [];
  public showPassword: boolean;

  @Input() public type: string | any;

  @Input() public fieldShow: boolean | any;

  @Input() public entity: FormInterface | any;

  @Input() public formGroup: UntypedFormGroup | any;

  @Input() public selectables: { [name: string]: FormSelectable };

  @Input() public suggestions: { [name: string]: (text: Observable<string>) => Observable<readonly any[]>; };

  @Input() public selectedCountryISO: CountryISO;

  @Input() public confirmSubmit: boolean | any;

  @Input() public errorMessage: string | any;

  @Input() alertRequired: boolean | any;

  /**
   * actionChangeLoginText
   * */
  @Output() public eventEmitterActionChangeLoginText: EventEmitter<any>;

  /**
   * actionChangeSimpleNumber
   * */
  @Output() public eventEmitterActionChangeSimpleNumber: EventEmitter<any>;

  /**
   * actionChangeSimpleText
   * */
  @Output() public eventEmitterActionChangeSimpleText: EventEmitter<any>;

  /**
   * actionChangeSimpleTextArea
   * */
  @Output() public eventEmitterActionChangeSimpleTextArea: EventEmitter<any>;

  /**
   * actionChangeSimpleSelector
   * */
  @Output() public eventEmitterActionChangeSimpleSelector: EventEmitter<any>;

  /**
   * actionChangeMultiSelector
   * */
  @Output() public eventEmitterActionChangeMultiSelector: EventEmitter<any>;

  /**
   * actionChangeSimpleDate
   * */
  @Output() public eventEmitterActionChangeSimpleDate: EventEmitter<any>;

  @Output() actionUploadImageForm = new EventEmitter<string>();

  @Output() actionUploadDocumentForm = new EventEmitter<string>();

  public preferredCountries: string[];
  now: Date = new Date();
  minDate = { year: 1900, month: 1, day: 1 };
  name: any;

  public defaultSuggestion: (text: Observable<string>) => Observable<readonly any[]>;

  constructor(
    protected readonly _formBuilder: UntypedFormBuilder,
  ) {
    this.eventEmitterActionChangeLoginText = new EventEmitter();
    this.eventEmitterActionChangeSimpleNumber = new EventEmitter();
    this.eventEmitterActionChangeSimpleText = new EventEmitter();
    this.eventEmitterActionChangeSimpleTextArea = new EventEmitter();
    this.eventEmitterActionChangeSimpleSelector = new EventEmitter();
    this.eventEmitterActionChangeMultiSelector = new EventEmitter();
    this.eventEmitterActionChangeSimpleDate = new EventEmitter();

    this.preferredCountries = [
      CountryISO.Nicaragua,
      CountryISO.Panama,
      CountryISO.Guatemala,
      CountryISO.UnitedStates
    ];

    this.selectedCountryISO = CountryISO.Nicaragua;

    this.selectables = {};
    this.suggestions = {};

    this.defaultSuggestion = (text: Observable<string>) => text.pipe(map(() => []));
  }



  private initializeForm(): void {
    this.segments?.forEach((segment) => {
      segment.templates.forEach((template) => {
        const validators: any[] = [];
        if (template.isRequired) validators.push(Validators.required);
        if (template.isEmail) validators.push(Validators.email);

        let value: any = this.entity[template.field];
        if (template.isDate) {
          let dateMoment = moment();
          if (value) {
            dateMoment = moment(value);
            value = {
              year: dateMoment.year(),
              month: dateMoment.month() + 1,
              day: dateMoment.date()
            };
          }
        }

        this.formGroup.removeControl(template.field);

        this.formGroup.addControl(
          template.field,
          new UntypedFormControl({
            value: value,
            disabled: template.isDisabled
          },
          validators));

      });
    });
  }

  public ngOnChanges(simpleChanges: SimpleChanges): void {
    this.segments = this.entity?.setFormSegment(this.type, this.fieldShow);
    try {
      if (
        simpleChanges &&
        simpleChanges['entity'] &&
        simpleChanges['entity'].currentValue
      ) {
        this.entity = simpleChanges['entity'].currentValue;
        this.segments = this.entity.setFormSegment(this.type, this.fieldShow);

        this.initializeForm();
      }
    } catch (exception) {
      console.log(exception);
    }
  }

  public actionSubmit(): void { }


  public actionChangeSimpleSelector(value: number, field: string): void {
    let index = 0;
    let selectable = null;
    if (value) {
      try {
        selectable = this.selectables[field].data.find((data: any) =>
          data[this.selectables[field].value] === value);
        if (selectable) {
          index = this.selectables[field].data.indexOf(selectable);
        }
      } catch (exception) {
        console.log(exception);
      }
    }
    this.eventEmitterActionChangeSimpleSelector.emit({ index, field, selectable, });
  }

  public actionChangeMultiSelector(selection: any[], field: string): void {
    this.eventEmitterActionChangeMultiSelector.emit({ selection, field });
  }


  public actionUploadImage(event: string | undefined) {
    this.actionUploadImageForm.emit(event);
  }

  ngOnInit(): void {
    this.initializeForm();
  }
}
