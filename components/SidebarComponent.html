import {Component, Input, OnInit} from '@angular/core';
import {Router} from "@angular/router";
import {SidebarStyleService} from '../../../pages/logistic/state/Observables/sidebar-style.service';
import { UserStateService } from 'src/app/state/Observables/user-state.service';
import { NotificationsService } from 'src/app/services/notifications.service';

@Component({
  selector: 'app-sidebar',
  templateUrl: './sidebar.component.html',
  styleUrls: ['./sidebar.component.scss']
})
export class SidebarComponent implements OnInit {
  @Input() idVendor: number;
  @Input() idStore: number;
  @Input() userManager: string;
  public amount: any = 0;

  constructor(
    public _router: Router,
    public _sidebarStyleService: SidebarStyleService,
    private _userStateService: UserStateService,
    private _notificationsService: NotificationsService
  ) { }

  public dropDown() {
    let submenu =  document.querySelector('#submenu') as HTMLElement;
    submenu.classList.toggle('hidden');

    let arrow = document.querySelector('#arrow') as HTMLElement;
    arrow?.classList.toggle('rotate-0');
  }

  public dropDownSettings() {
    let submenuSettings =  document.querySelector('#submenuSettings') as HTMLElement;
    submenuSettings.classList.toggle('hidden');
  }

  ngOnInit(): void {

    const body = document.querySelector('body') as HTMLElement,
      sidebar = body.querySelector('aside') as HTMLElement,
      toggle = body.querySelector(".toggle") as HTMLElement;

    toggle.addEventListener("click" , () =>{
      sidebar.classList.toggle("close");
      this._sidebarStyleService.setChangeVisibleObservable();
    });

    this._userStateService.user$.subscribe({
      next: (response: any) => {
        this.userManager = response.user.username;
        this._notificationsService.notifyQuantityDeliveries(this.userManager);
      }});

    this._notificationsService.amount.subscribe({
      next: (response: any) => {this.amount = response;},
      error: (err) => {}
    });
  }
}
