import { Component, OnInit } from '@angular/core';
import {TizoListComponent} from '../../../../../../schemes/abstracts/tizo-list.component';
import {Router} from '@angular/router';
import { UserService } from "../../../../../../services/user/user.service";
import { plainToInstance } from "class-transformer";
import { ResponseRequest } from "../../../../../../schemes/entities/response/response-request";
import { NgxSpinnerService } from "ngx-spinner";
import {StoreService} from "../../../../state/store.service";
import {UserStateService} from "../../../../../../state/Observables/user-state.service";
import {switchMap, tap} from "rxjs";

@Component({
  selector: 'app-users',
  templateUrl: './users.component.html',
  styleUrls: ['./users.component.scss']
})
export class UsersComponent extends TizoListComponent<any>  implements OnInit {


  public selectedType: string;
  public typeUserManager: any[];
  public userState: any;
  public idUserVendor: number | any;

  public override entities: any[];

  constructor(
    protected override readonly _router: Router,
    private _spinner: NgxSpinnerService,
    private _restUser: UserService,
    private _stateStore: StoreService,
    private _userService: UserStateService,
  ) {
    super(_router);
    this.typeUserManager = [
      {key: 'vendor', value: 'Vendedor'},
    ];
  }

  public ngOnInit(): void {
    void this._spinner.show();
    this.loadData();
  }

  public override loadData(): void {
    this._restUser.getUserData().pipe(
      switchMap(({data}: any) => this._restUser.getUsersVendor(data.user.storesPermissions[0].idStore))
    ).subscribe({
      next: (response) => {

        let {data} = plainToInstance(ResponseRequest, response);
        this.entities = data?.users.filter((item: any)=>item.idStatusVendorUser === 0);
        this._spinner.hide();
      },
      error: (err) => {
        console.log(err);
        this._spinner.hide();
      }
    });
  }

  public override actionEdit(data: any) {
    super.actionEdit(data);
    this._router.navigate([`/home/vendor/management/access/user/${data.idVendorUser}`]);
  }

  public override actionRemove(data: any) {
    this.isShowQuestionDialog = true;
    this.idUserVendor = data.idVendorUser;
    this.messageDialogQuestion = `Est seguro de eliminar el usuario ${data.username}?`;
  }

  public actionDialogResponse(response: boolean): void {
    this._restUser.deleteVendorUser(this.idUserVendor).subscribe({
      next: (response) => {
        this.loadData();
      },
      error: (err) => {}
    });
  }

  public actionDialogNo(){
    this.isShowQuestionDialog = false;
  }

  public override actionSave() {
    super.actionSave();
    this._router.navigate([`/home/vendor/management/access/user`]);
  };


}
