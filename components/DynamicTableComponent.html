import {
  AfterViewInit,
  Component,
  ContentChild,
  EventEmitter,
  Input,
  OnInit,
  Output,
  ViewEncapsulation
} from '@angular/core';
import {TableTemplate} from "../../abstract/tableTemplate";
import {TizoListComponent} from "../../abstract/tizo-list.component";
import {Router} from "@angular/router";
import {CdkDragDrop, moveItemInArray} from "@angular/cdk/drag-drop";
import {TizoTableService} from "../../services/tizo-table.service";

@Component({
  selector: 'app-dynamic-table',
  templateUrl: './dynamic-table.component.html',
  styleUrls: ['./dynamic-table.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class DynamicTableComponent<T> extends TizoListComponent<T> implements OnInit, AfterViewInit {

  @ContentChild(Component) columnCustomComponent: Component;

  public headers: string[];

  keyWordSearch: string = '';

  @Input() public override entity: T;

  /**
   * @description title of table
   * */
  @Input()  public override title: string;

  /**
   * @description status of the data
   * */
  @Input()  public override status: number;

  /**
   * @description table data
   * */
  @Input()  public override entities: any[];

  /**
   * @description types template
   * */
  @Input() public override templates: TableTemplate[];

  /**
   * @description event emitter for excel export
   * */
  @Input()  public override customizeExcelCell: any;

  /**
   * @description showCheckbox mode
   * */
  @Input()  public showCheckBoxesMode: boolean = true;

  @Input()  public hasButtonView: boolean;

  @Input()  public hasButtonViewDelivery: boolean;

  @Input()  public hasButtonViewDriver: boolean;

  @Input()  public hasButtonUpload: boolean;

  @Input()  public hasButtonFileUpload: boolean;

  @Input()  public hasButtonEdit: boolean;

  @Input()  public hasButtonShipment: boolean;

  @Input()  public hasButtonRemove: boolean;

  @Input()  public hasButtonClean: boolean;

  @Input()  public hasButtonRestore: boolean;

  @Input()  public typeEntity: string;

  @Output() public eventEmitterOnContentReady: EventEmitter<any>;

  @Output() public eventEmitterOnCellPrepared: EventEmitter<any>;

  @Output() public eventEmitterOnToolbarPreparing: EventEmitter<any>;

  @Output() public eventEmitterOnSelectionChanged: EventEmitter<any>;

  @Output() public eventEmitterOnSelectionRowChanged: EventEmitter<any>;

  @Output() public eventEmitterActionMap: EventEmitter<any>;

  @Output() public eventEmitterActionView: EventEmitter<any>;

  @Output() public eventEmitterActionViewDriver: EventEmitter<any>;

  @Output() public eventEmitterActionViewDelivery: EventEmitter<any>;

  @Output() public eventEmitterActionUpload: EventEmitter<any>;

  @Output() public eventEmitterActionEdit: EventEmitter<any>;

  @Output() public eventEmitterActionShipment: EventEmitter<any>;

  @Output() public eventEmitterActionRemove: EventEmitter<any>;

  @Output() public eventEmitterActionClean: EventEmitter<any>;

  @Output() public eventEmitterActionRestore: EventEmitter<any>;



  public showWarningDialog: boolean;

  public file:any;

  public selectedRows: any[];

  //*=====================================================*//
  public current: number = 1;
  //public total: number = 20;

  public onGoTo(page: number): void {
    this.current = page;
    this.itemsToDisplay = this.paginate(this.current, this.perPage);
  }


  public onNext(page: number): void {
    this.current = page + 1;
    this.itemsToDisplay = this.paginate(this.current, this.perPage);
  }

  public onPrevious(page: number): void {
    this.current = page - 1;
    this.itemsToDisplay = this.paginate(this.current, this.perPage);
  }

  /*=========================================================*/
  /*=======================Example===========================*/

  public itemsToDisplay: any[] = [];

  public perPage = 10;
  public total = Math.ceil(this.entities.length / this.perPage);


  public paginate(current: number, perPage: number): string[] {
    return [...this.entities.slice((current - 1) * perPage).slice(0, perPage)];
  }

  /*=========================================================*/

  constructor(
    protected override _router: Router,
    private _tizoTableService: TizoTableService
  ) {
    super(_router);
    this.eventEmitterOnContentReady = new EventEmitter();
    this.eventEmitterOnCellPrepared = new EventEmitter();
    this.eventEmitterOnToolbarPreparing = new EventEmitter();
    this.eventEmitterOnSelectionChanged = new EventEmitter();
    this.eventEmitterOnSelectionRowChanged = new EventEmitter();
    this.eventEmitterActionMap = new EventEmitter();
    this.eventEmitterActionView = new EventEmitter();
    this.eventEmitterActionViewDriver = new EventEmitter();
    this.eventEmitterActionViewDelivery = new EventEmitter();
    this.eventEmitterActionUpload = new EventEmitter();
    this.eventEmitterActionEdit = new EventEmitter();
    this.eventEmitterActionShipment = new EventEmitter();
    this.eventEmitterActionClean = new EventEmitter();
    this.eventEmitterActionRemove = new EventEmitter();
    this.eventEmitterActionRestore = new EventEmitter();
    this.showWarningDialog = false;
    this.showCheckBoxesMode = false;
  }

  public ngOnInit(): void {
    this.itemsToDisplay = this.paginate(this.current, this.perPage);
  }

  public ngAfterViewInit(): void {
    /// console.log(this.entities);
  }


  public override actionOnSelectedRows(event: any): void {
    this.eventEmitterOnSelectionChanged.emit(event);
  }

  public actionOnSelectedRow(event: any): void {
    this.eventEmitterOnSelectionRowChanged.emit(event.key);
  }

  public override actionView(data: any): void {
    this.eventEmitterActionView.emit(data);
  }


  public override actionUpload(event: any, data: any): void {

  }

  public override actionEdit(data: any): void {
    this.eventEmitterActionEdit.emit(data);
  }

  public override actionShipment(data: any): void {
    this.eventEmitterActionShipment.emit(data);
  }

  public override actionRemove(data: any): void {
    this.eventEmitterActionRemove.emit(data);
  }

  public detectFiles($event: any) {
    const files = $event.target.files[0];

    const reader = new FileReader();
    reader.onload = (e: any) => {
      const name = files.name;
      const extension = name.split('.').pop();
      const document = {
        name: name,
        extension: extension,
        url: e.target.result,
        path: '',
        description: '',
        base64: e.target.result,
      };
      this.file = document;
      this.eventEmitterActionUpload.emit(document);
    };

    reader.readAsDataURL(files);
  }

  public onRowPrepared(event$: any): void {
    ///console.log("esto es values: ", event$?.values[4]);
    let {rowElement} = event$;

    // 6,Claim

      'Orden en distribucin',
      'Orden en camino',
    ];

    if(event$.rowType === 'data'){
      if(event$.values[5] === 0){
        rowElement.style.backgroundColor = 'rgba(253,255,255,0.78)';
      }else if(step.includes(event$.values[6])){
        rowElement.style.backgroundColor = 'rgba(253,152,3,0.07)';

      }else if(event$.values[5] === 3){
