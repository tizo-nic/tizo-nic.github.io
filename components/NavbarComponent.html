import { Component, EventEmitter, OnInit, Output } from "@angular/core";
import {Router} from '@angular/router';
import {plainToInstance} from 'class-transformer';
import { NgxSpinnerService } from "ngx-spinner";

import {AuthService} from '../../../auth/auth.service';
import {UserStateService} from '../../../state/Observables/user-state.service';
import { TizoStorageService } from "../../../services/tizo-storage.service";
import {IStoreModel} from "../../../pages/vendor-manager/models/store.model";
import {StoreService} from "../../../pages/vendor-manager/state/store.service";
import {BranchRestService} from "../../../pages/vendor-manager/services/branch-rest.service";
import {IBranchModel} from "../../../pages/vendor-manager/models/branch.model";
import {StoreEntity} from "../../../pages/vendor-manager/entities/store.entity";
import {forkJoin, Observable, switchMap, tap} from "rxjs";
import {ResponseRequest} from "../../../schemes/entities/response/response-request";
import {BranchService} from "../../../pages/vendor-manager/state/branch.service";
import {UserService} from "../../../services/user/user.service";
import {AdminSystemService} from "../../../services/admin-system/admin-system.service";
import {TypeSystem} from "../../../schemes/enum/type-system";


@Component({
  selector: 'app-navbar',
  templateUrl: './navbar.component.html',
  styleUrls: ['./navbar.component.scss'],
})
export class NavbarComponent implements OnInit {

  public nameUser: string;
  public idUserVendor: number;
  public typeUser: string;
  public isShowPopupPassword: boolean = false;
  public pathImage: string;
  public storeEntity: IStoreModel = new StoreEntity();
  public branches: IBranchModel[] = [];

  public isSuccessVisible = false;
  public successMessage: string;

  public environment: string;
  public isProduction: boolean = true;

  public isShowWarningDialog: boolean = false;

  public isWarningVisible = false;
  public warningMessage: string;

  @Output() changeEnvironment = new EventEmitter<any>();

  public branchIndex: number = 0;
  public isDemo = false;
  public typeSystem$: Observable<TypeSystem>;

  constructor(
    protected _router: Router,
    private _authService: AuthService,
    private _userStateService: UserStateService,
    private _tizoStorageService: TizoStorageService,
    private _spinner: NgxSpinnerService,
    private _stateStore: StoreService,
    private _branchRestService: BranchRestService,
    private _branchStore: BranchService,
    private _restUser: UserService,
    private _tizoStorage: TizoStorageService,
    private _adminSystemService: AdminSystemService,
  ) {
    this.isDemo = _tizoStorage.getEnviorment() === 'dev';
    this.typeSystem$ = _adminSystemService.getSystem$;
  }



  public ngOnInit() {

    this.environment = this._tizoStorageService.getDevEnvironmentNic();

    this.environment === 'dev' ? this.isProduction = false : this.isProduction = true;

    this._stateStore.store$.pipe(
      tap((store)=>this.storeEntity = store),
      switchMap(( store:any) =>
        forkJoin([
          this._branchRestService.getAllBranches(store.idStore),
          this._restUser.getUserData()
        ])),
    ).subscribe({
      next: ([responseBranch, responseUser]) => {
        let {data: dataResponseBranch} = plainToInstance(ResponseRequest, responseBranch);
        let {data: dataResponseUser} = plainToInstance(ResponseRequest, responseUser);

        this.branches  = dataResponseBranch?.branches.filter((branch:any) => branch.active);
        let branchesPermissions = dataResponseUser.user.storeBranchesPermissions;

        this.branchIndex = (this._tizoStorageService.getindexLastBranchSelected());

        let newBranches: any[] = [];

        // filtrar las branches que el usuario tiene permisos
        branchesPermissions.map((branch: any)=>{
          this.branches.map((item)=>{
            if(item.idStoreBranch === branch.idStoreBranch){
              newBranches.push(item);
            }
          });
        });


        this.branches = newBranches;
        this.branches.length < this.branchIndex ? this.branchIndex = 0 : null;
        // // Cargar la sucursal del usuario por defecto
        this._branchStore.addBranch(this.branches[this.branchIndex]);
        this.nameUser = dataResponseUser.user?.username || '...';
        this.typeUser = dataResponseUser.user.type_user || '...';
        this.pathImage = dataResponseUser.user?.pathImage;
        this.idUserVendor = (this.typeUser === 'vendor_user') ? dataResponseUser.user?.idVendorUser : -1;
        this._spinner.hide();
      },
    });

  }

  public handleCloseSession(){
    this._authService.authSignOut();
    this._authService.authSignOutEnvironment();
    void this._router.navigateByUrl('/signin');
  }

  public handleNavigateToProfile(){
    this._router.navigate(['/home/vendor/management/access/users/']);
  }

  public changePassword(){
    this.isShowPopupPassword = true;
  }

  public actionChangeBranch(branch: any, index: number){
    this.branchIndex = index;
    this._branchStore.addBranch(branch);
    this._tizoStorageService.setIndexLastBranchSelected(index);
  }

  public eventEmitterClose(){
    this.isSuccessVisible = false;
    this.isWarningVisible = false;
  }

  public actionOk(){
    this.isSuccessVisible = false;
    this.isWarningVisible = false;
  }

  public onMessageVisible(message: any) {
    //cerrar popup de cambiar contraseña
    this.isShowPopupPassword = false;
    this.isShowWarningDialog = message.isWarningVisible;

    //activar popup success o warning
    this.isSuccessVisible = message.isSuccessVisible;
    this.isWarningVisible = message.isWarningVisible;

    //poner mensaje de popup success o warning
    this.successMessage = message.message;
    this.warningMessage = message.message;
  }

}
