import { Component, OnInit, EventEmitter, Output } from '@angular/core';
import { NgxSpinnerService } from "ngx-spinner";
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from "@angular/forms";
import { Router } from "@angular/router";
import { UserService } from "../../services/user/user.service";
import { TizoEditorComponent } from "../../schemes/abstracts/tizo-editor.component";
import { ChangePasswordEntity } from "../../schemes/entities/change-password.entity";

@Component({
  selector: 'app-change-password',
  templateUrl: './change-password.component.html',
  styleUrls: ['./change-password.component.scss']
})
export class ChangePasswordComponent extends TizoEditorComponent<ChangePasswordEntity>  {

  @Output() isMessageVisible = new EventEmitter<any>();

  constructor(
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _router: Router,
    private _userService: UserService
  ) {
    super(_spinner, _formBuilder, _router);
  }


  public changePassword = new UntypedFormGroup({
    currentPassword: new UntypedFormControl('', Validators.required),
    newPassword: new UntypedFormControl('', Validators.required),
    newPasswordConfirm: new UntypedFormControl('', Validators.required),
  });

  public onSubmit(){

    let postResetPassword = {
      'password': this.changePassword.get('newPassword')?.value,
      'oldPassword': this.changePassword.get('currentPassword')?.value
    };

    if (this.changePassword.get('newPassword')?.value !== this.changePassword.get('newPasswordConfirm')?.value){

      let message = {
        'message': 'Nueva contrasea y confirmar contrasea deben coincidir',
        'isSuccessVisible': false,
        'isWarningVisible': true
      };

      this.isMessageVisible.emit(message);
      return;
    }

    this._userService.postResetPassword(postResetPassword).subscribe({
      next: (data) => {

        let message = {
          'message': 'Se han actualizado las credenciales de acceso',
          'isSuccessVisible': true,
          'isWarningVisible': false
        };

        this.isMessageVisible.emit(message);
      },
      error: (err) => {

        let message = {
          'message': 'Actual contrasea incorrecta',
          'isSuccessVisible': false,
          'isWarningVisible': true
        };

        this.isMessageVisible.emit(message);
      }
    });

    //limpiar formulario despues de haberlo enviado
    this.changePassword.reset();
  }
}
