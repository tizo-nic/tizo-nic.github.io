  public formGroupModules: UntypedFormGroup;


  constructor(protected override readonly _spinner: NgxSpinnerService,
              protected override readonly _formBuilder: UntypedFormBuilder,
              protected override readonly _router: Router,
              private _activeRoute: ActivatedRoute,
              private _formBuilderModules: UntypedFormBuilder,
              private _restProfileManager: ProfilevendorRestService
  ) {
    super(_spinner, _formBuilder, _router);
    this.formGroupModules = this._formBuilderModules.group({
      modules: this._formBuilderModules.array([])
    });
  }

  public get modules(): UntypedFormArray {
    return this.formGroupModules.get('modules') as UntypedFormArray;
  }

  public newModule(): UntypedFormGroup {
    return this._formBuilderModules.group({
      'module': false,
    });
  };

  public addModule(): void {
    this.modules.push(this.newModule());
  }


  ngOnInit(): void {
    this.entity = new ProfileManagerEntity();
    this.idElement = this._activeRoute.snapshot.paramMap.get('idProfile') as string;
    this.loadData();
    if(this.idElement){
      this.initializeFormEditor();
    }
  }

  public override loadData(): void{

    // this._restProfileManager.getDetailsPermissionsModulesVendor().subscribe({
    //   next: (response) => {
    //     let {data} = plainToInstance(ResponseRequest, response);
    //     this.modulesPermissions = data.modules;
    //   },
    //   error: (error: any) => {
    //   }
    // });

    this.modulesPermissions?.map(()=>{
      this.addModule();
    });
  }

  public initializeFormEditor(): void {
    void this._spinner.show();
    this._restProfileManager.getProfileVendorById(this.idElement).subscribe({
      next: (responseProfile) => {
        let {data} = plainToInstance(ResponseRequest, responseProfile);
        this.entity = plainToInstance(ProfileManagerEntity, data.profileVendorUser);

        // asignar true a los modulos que tiene el usuario del formGroupArray
        data.accessTo.map((module: any)=>{
          this.modulesPermissions.map((modulePermission: any, i: number)=>{
            if(module.idVendorModule === modulePermission.idModule){
              this.modules.controls[modulePermission.idModule].get('module')!.setValue(true);
            }});
        });
        this._spinner.hide();
      },
      error: (error) => {
        console.log(error);
        void this._spinner.hide();
      }
    });
  };


  public override actionSave() {
    super.actionSave();
    // filter switchValue when state is false
    this.switchValue = this.switchValue.filter((item: any) => item.state === true);
    let data = this.formGroup.getRawValue();
    let {modules} = this.formGroupModules.getRawValue();

    let test = this.modulesPermissions.map((item: any, index: number) => {
      if(modules[index].module === true){
        return item;
      }
    });

    test = test.filter((item: any) => item !== undefined);


    let dataRequest = {
      "name": data.name ?? 'user default',
      "description": data.data ?? 'description default',
      "modulePermissions": test
    };

    this._restProfileManager.postProfileVendor(dataRequest).subscribe({
      next: (response) => {
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'Perfil creado correctamente';
      },
      error: (error) => {
      }
    });
  }



  public override actionUpdate() {
    this.switchValue = this.switchValue.filter((item: any) => item.state === true);
    let data = this.formGroup.getRawValue();
    let {modules} = this.formGroupModules.getRawValue();

    let test = this.modulesPermissions.map((item: any, index: number) => {
      if(modules[index].module === true){
        return item;
      }
    });
    test = test.filter((item: any) => item !== undefined);
    let dataRequest = {
      "name": data.name ?? 'user default',
      "description": data.data ?? 'description default',
      "modulePermissions": test
    };

    this._restProfileManager.putProfileVendor(this.idElement,dataRequest).subscribe({
      next: (response) => {
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'Perfil actualizado correctamente';
      },
      error: (error) => {
      }
    });
  }

  public override actionCancel(): void {
    this._router.navigate(['/home/vendor/management/access/profiles-vendor']);
  }


  public override actionOkDialogSuccess(): void {
    this.isShowSuccessDialog = false;
    this._router.navigate(['home/vendor/management/access/profiles-vendor']);
  }

}
