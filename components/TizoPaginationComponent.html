import {
  Component,
  EventEmitter,
  Input,
  OnChanges,
  Output,
  SimpleChanges
} from '@angular/core';

@Component({
  selector: 'app-tizo-pagination',
  templateUrl: './tizo-pagination.component.html',
  styleUrls: ['./tizo-pagination.component.scss']
})
export class TizoPaginationComponent implements OnChanges {

  /**
   * @description - muestra la pagina actual que se esta visualizando
   * */
  @Input() current: number = 1;

  /**
   * @description - muestra la cantidad de paginas que se van a mostrar
   */
  @Input() total: number = 0;

  @Input() sizePage: number[];

  @Output() goTo: EventEmitter<number> = new EventEmitter<number>();

  @Output() next: EventEmitter<number> = new EventEmitter<number>();

  @Output() previous: EventEmitter<number> = new EventEmitter<number>();

  /*==================================================================*/

  public pages: number[] = [];

  constructor() {
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (
      (changes['current'] && changes['current'].currentValue) ||
      (changes['total'] && changes['total'].currentValue)
    ) {
      this.pages = this.getPages(this.current, this.total)
    }
  }

  public onGoTo(page: number): void {
    this.goTo.emit(page);
  }

  public onNext(): void {
    this.next.emit(this.current);
  }

  public onPrevious(): void {
    this.previous.emit(this.current);
  }

  public getPages(current: number, total: number): number[] {
    if (total <= 7) {
      return [...Array(total).keys()].map(x => ++x);
    }

    if (current > 5) {
      if (current >= total - 4) {
        return [1, -1, total - 4, total - 3, total - 2, total - 1, total];
      } else {
        return [1, -1, current - 1, current, current + 1, -1, total];
      }
    }

    return [1, 2, 3, 4, 5, -1, total];
  }

  public onCurrentSizePage(sizePage: number): void {
    //this.currentSizePage.emit(sizePage);
  }


}
