import {Component, ElementRef, EventEmitter, Input, Output, ViewChild} from "@angular/core";
import {ImageTransform } from "ngx-image-cropper";

@Component({
  selector: 'app-image-cropper',
  templateUrl: './image-cropper.component.html',
  styleUrls: ['./image-cropper.component.scss']
})
export class ImageCropperComponent  {

  public file: File;

  @Input() nameImage: string;

  @Input() nameAvatar: string;

  @Input() pathImage: string;

  @Input() ngClass: any;

  @Output() actionUploadImage = new EventEmitter<any>();
  @ViewChild('imageFileInput') imageFileInput: ElementRef;

  public imageChangedEvent: any;
  public croppedImage: any;
  public imageUrl: any;

  public canvasRotation: number;
  public rotation: number;
  public scale: number;
  public aspectRatio: number;
  public imageTransform: ImageTransform;
  public containWithinAspectRatio: boolean;

  public isEditMode: boolean;
  public constants: any;
  public document: any;

  constructor() {
    this.imageChangedEvent = '';
    this.croppedImage = '';
    this.canvasRotation = 0;
    this.rotation = 0;
    this.scale = 1;
    this.aspectRatio = 1;
    this.imageTransform = {};
    this.containWithinAspectRatio = false;
    this.isEditMode = false;
    this.constants = null;
    this.document = null;
  }


  public detectFiles($event: any) {
    this.file = $event.target.files[0];
    const reader = new FileReader();
    reader.onload = (e: any) => {
      const name = this.file.name;
      const extension = name.split('.').pop();
      this.imageUrl = {
        name: name,
        extension: extension,
        url: e.target.result,
        path: '',
        description: '',
        base64: e.target.result,
      };

      this.croppedImage = this.imageUrl.base64;
    };
    reader.readAsDataURL(this.file);
  }



  public saveImage() {
    this.actionUploadImage.emit(this.file);
  }

  public deleteImage() {
    this.imageUrl = null;
    this.croppedImage = null;
  }

}
