import {Component, Input, OnInit} from '@angular/core';
import {Router} from "@angular/router";
import {SidebarStyleService} from "../../../pages/logistic/state/Observables/sidebar-style.service";

@Component({
  selector: 'app-sidebar-sales',
  templateUrl: './sidebar-sales.component.html',
  styleUrls: ['./sidebar-sales.component.scss']
})
export class SidebarSalesComponent implements OnInit {

  @Input() idVendor: number;
  @Input() idStore: number;
  @Input() userManager: string;

  constructor(
    public _router: Router,
    public _sidebarStyleService: SidebarStyleService,
  ) { }

  ngOnInit(): void {

    const body = document.querySelector('body') as HTMLElement,
      sidebar = body.querySelector('aside') as HTMLElement,
      toggle = body.querySelector(".toggle") as HTMLElement;

    toggle.addEventListener("click" , () =>{
      sidebar.classList.toggle("close");
      this._sidebarStyleService.setChangeVisibleObservable();
    });

  }

}
