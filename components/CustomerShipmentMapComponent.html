import {
  Component,
  ElementRef,
  EventEmitter,
  Input,
  NgZone, OnChanges, OnInit,
  Output,
  SimpleChanges,
  ViewChild
} from "@angular/core";
import {TizoMapEditorComponent} from "../../schemes/abstracts/tizo-map-editor.component";
import {NgxSpinnerService} from "ngx-spinner";
import {UntypedFormBuilder} from "@angular/forms";
import {MapsAPILoader} from "@agm/core";
import {ActivatedRoute, Router} from "@angular/router";
import {MapService} from "../../services/map.service";
import {WaypointService} from "../../pages/logistic/state/Observables/waypoint.service";
import {CustomerService} from "../../pages/customer/customer.service";
import {PositionCustomerService} from "../../pages/logistic/state/Observables/position-customer.service";

declare const google: any;

@Component({
  selector: 'app-customer-shipment-map',
  templateUrl: './customer-shipment-map.component.html',
  styleUrls: ['./customer-shipment-map.component.scss']
})
export class CustomerShipmentMapComponent extends TizoMapEditorComponent<any> implements OnChanges, OnInit{

  @ViewChild('search')
  public searchElementRef: ElementRef;

  @Input() override longitude: number;

  @Input() override latitude: number;

  @Input() override zoom: number;

  /**
   * Verifica si el punto se va a editar
   * */
  @Input() editWayPoint: any;

  @Input() searchPlaces: string;

  @Output() selectPoint = new EventEmitter<any>();

  public itemsPoint: any[] = [];

  public idCustomer: string;

  public idStore: string;

  constructor(
    protected readonly _positionCustomerService: PositionCustomerService,
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _mapsAPILoader: MapsAPILoader,
    protected override readonly _router: Router,
    protected override readonly _ngZone: NgZone,
    protected override readonly _mapService: MapService,
    private _wayPointService: WaypointService,
    private _customerService: CustomerService,
    private _activeRoute: ActivatedRoute
  ) {
    super(_spinner, _formBuilder, _mapsAPILoader, _router, _ngZone, _mapService);
    this.zoom = 11;
  }

  public ngOnInit() {

    this.idCustomer = this._activeRoute.snapshot.paramMap.get('idCustomer') as string;
    this.idStore = this._activeRoute.snapshot.paramMap.get('idStore') as string;

    this._positionCustomerService.position$.subscribe({
      next: (position) => {
        void this.selectedPoints(position.latitude, position.longitude);
      }
    });
  }

  public ngOnChanges(simpleChanges: SimpleChanges): void {
    try {
      if (simpleChanges && simpleChanges['zoom'] && simpleChanges['zoom'].currentValue) {
        this.zoom = simpleChanges['zoom'].currentValue;
      }

      if (simpleChanges && simpleChanges['searchPlaces'] && simpleChanges['searchPlaces'].currentValue) {
        this.searchPlaces = simpleChanges['searchPlaces'].currentValue;
      }

    } catch (exception) { }
  }

  public override onChangeLanguage() {
    super.onChangeLanguage();
  }

  public override loadData(): void {
    super.loadData();
  }

  /**
   * @description - Método para establecer un punto X en el mapa
   * con base en una latitud y longitud
   * */
  public async placeMarketPoint($event: any): Promise<void> {
    let {coords} = $event;
    await this.selectedPoints(coords.lat, coords.lng);
  }

  /**
   * @description - Método para establecer una imagen PNG en el mapa
   * */
  public async selectedPoints(lat: number = 0.0, lng: number = 0.0) {
    let img = 1;
    let url = {
      url: `assets/img/logistic/maps/${this.getLetter(img)}-icon.png`,
      scaledSize: {
        width: 40,
        height: 40
      }
    };
    await this.getAddress(lat, lng, url);
  }

  /**
   * @description - Método para obtener la dirección de una ubicación
   * @param lat - Latitud
   * @param lng - Longitud
   * @param url - Url de la imagen
   * */
  public async getAddress(lat: number, lng: number, url: any): Promise<void> {
    let addressStreet = '', address = '', country = '', state = '', city = '';

    let geoCoder = new google.maps.Geocoder();
    let latLng = new google.maps.LatLng(lat, lng);
    const request: google.maps.GeocoderRequest = {
      location: latLng
    };

    await geoCoder.geocode(request, (results: string | any[], status: any) => {
      if (results !== null) {
        address = results[0].formatted_address;
        const length = results.length;
        addressStreet = results[length - 4]?.formatted_address.split(',')[0];
        country = results[length - 1]?.formatted_address;
        state = results[length - 2]?.formatted_address.split(',')[0];
        city = results[length - 3]?.formatted_address.split(',')[0];
        this.addPoints(lat, lng, addressStreet, address, url, country, state, city);
      }
    });
  }

  /**
   * @description - Método para agregar un punto al mapa
   * @param lat - Latitud
   * @param lng - Longitud
   * @param addressStreet - Dirección
   * @param address - Dirección
   * @param url - Url de la imagen
   * @param country - País
   * @param state - Estado
   * @param city - Ciudad
   * */
  public addPoints(lat: number, lng: number,addressStreet: string, address: string, url: any, country: string, state: string, city: string){

    // Orden del proceso
    // 1. Agregar el punto A al mapa
    //   a. Verificar si el punto ya existe
    //   b.

    // 2. Agregar el punto B al mapa
    //   a. Verificar si el punto ya existe

    let point = {
      latitude: lat,
      longitude: lng,
      addressGeo: address,
      addressStreet: addressStreet,
      url: url,
      city: city,
      state: state,
      country: country
    };

    this.itemsPoint[0] = point;
    this.selectPoint.emit(point);

    this.latitude = lat;
    this.longitude = lng;

  }

  /**
   * @description - Método para obtener una ubicación
   *
   * */
  public searchAddress() {
    const autoComplete = new google.maps.places.Autocomplete(this.searchElementRef?.nativeElement);
    autoComplete.addListener('place_changed', () => {
      this._ngZone.run(() => {
        let place: google.maps.places.PlaceResult = autoComplete.getPlace();
        if (place.geometry === undefined || place.geometry === null) {
          return;
        }
        void this.selectedPoints(place.geometry.location.lat(), place.geometry.location.lng());
      });
    });
  }

  /**
   * @description - Método para obtener una letra
   * @param num - Índice
   * */
  public getLetter(num: number): string {
    let alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
    return alphabet[num];
  }

  protected override initializeMap(): void {
    super.initializeMap();
  }

  override placesMap() {
    super.placesMap();
    this.searchAddress();
  }


}
