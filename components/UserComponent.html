import { Component, OnInit } from '@angular/core';
import { TizoEditorComponent } from "../../../../../../schemes/abstracts/tizo-editor.component";
import { NgxSpinnerService } from "ngx-spinner";
import { UntypedFormBuilder } from "@angular/forms";
import { ActivatedRoute, Router } from "@angular/router";
import { UserService } from "../../../../../../services/user/user.service";
import { ResponseRequest } from "../../../../../../schemes/entities/response/response-request";
import { plainToInstance } from "class-transformer";
import { FormSelectable } from "../../../../../../schemes/support/form-selectable";
import { MapService } from "../../../../../../services/map.service";
import { HttpClient} from "@angular/common/http";
import {UserVendorEntity} from "../../../../../../schemes/entities/user-vendor.entity";
import {StoreService} from "../../../../state/store.service";
import {BranchRestService} from "../../../../services/branch-rest.service";
import {BranchService} from "../../../../state/branch.service";
import {UserStateService} from "../../../../../../state/Observables/user-state.service";


@Component({
  selector: 'app-user',
  templateUrl: './user.component.html',
  styleUrls: ['./user.component.scss']
})
export class UserComponent extends TizoEditorComponent<any> implements OnInit {

  typeUser: string;
  public password: string;
  public responseUser: any;
  public override ipUser: string = '';
  public store: any;
  public userState: any;

  public override isShowSuccessDialog: boolean = false;

  constructor(
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _router: Router,
    private _activatedRoute: ActivatedRoute,
    private _userService: UserStateService,
    private _mapService: MapService,
    private _restUser: UserService,
    public override _http: HttpClient,
    private _stateStore: StoreService,
    private _branchRestService: BranchRestService,
    private _branchStore : BranchService
  ) {
    super(_spinner, _formBuilder, _router);

    this.responseUser = {

    };
  }

  ngOnInit(): void {
    this.idElement = this._activatedRoute.snapshot.paramMap.get('idVendorUser') as string;
    this.loadData();
    this.loadProfileVendorUser();
    this.loadSelectableGender();
  }

  public override loadData(): void {
    // Get route link direction
    this.entity = new UserVendorEntity();
    this._restUser.getUserData().subscribe({
      next: ({data}:any) => {
        this.userState = data.user;
        let {idStore} = data.user.storesPermissions[0];

        this.loadDataBranches(idStore);
        if (this.idElement) {
          this.initializedFormEditor(idStore);
        }
      }
    });

    this._http.get('https://api.ipregistry.co/?key=pgkqpiif3nmlbdh0').subscribe({
      next: (response: any) => {
        this.ipUser = response.ip;
      }
    });
  }



  public loadProfileVendorUser = (): void =>{
    this._restUser.getProfilesVendor().subscribe({
      next: (response) => {
        let {data} = plainToInstance(ResponseRequest, response);
        const profiles: [] = data.profileVendorUser;
        const formSelectable: FormSelectable = new FormSelectable();
        formSelectable.value = 'idProfileVendorUser';
        formSelectable.name = 'name';
        formSelectable.data = profiles;
        this.selectables = this._mapService.extend(
          this._mapService.copy(this.selectables),
          {idProfileVendorUser: formSelectable}
        );
      },
      error: (error) => {}
    });
  };

  public loadDataBranches(idStore: any): void{
    this._branchRestService.getAllBranches(idStore).subscribe({
      next: (branches) => {
        let {data} = plainToInstance(ResponseRequest, branches);
        const formSelectable: FormSelectable = new FormSelectable();
        formSelectable.value = 'idStoreBranch';
        formSelectable.name = 'name';
        formSelectable.data = data.branches;
        this.selectables = this._mapService.extend(
          this._mapService.copy(this.selectables),
          {branches: formSelectable}
        );
      },
    });
  }

  public loadSelectableGender(): void{
    const formSelectableGender: FormSelectable = new FormSelectable();
    formSelectableGender.value = 'gender';
    formSelectableGender.name = 'name';
    formSelectableGender.data = [
      { name: 'Masculino', value: 1 },
      { name: 'Femenino', value: 2 },
      { name: 'Otro', value: 3 },
    ];
    this.selectables = this._mapService.extend(
      this._mapService.copy(this.selectables),
      { gender: formSelectableGender, }
    );
  }

  public loadGender = (type: string): number =>{
    switch (type){
    case 'Masculino':
      return 1;
    case 'Femenino':
      return 2;
    default:
      return 3;
    }
  };

  public initializedFormEditor(idStore: any): void{
    this._restUser.getVendorUserProfileById( idStore, this.idElement).subscribe({
      next: (response) => {
        let {data} = plainToInstance(ResponseRequest, response);
        data.user.phone = data.user.phone.slice(4);
        this.entity = plainToInstance(UserVendorEntity, data.user);
      },
      error: error => {
        console.log(error);
      }
    });
  }

  public override actionSave() {
    super.actionSave();
    this._spinner.show();
    const vendorUser = this.formGroup.getRawValue();
    vendorUser.gender = this.loadGender(vendorUser.gender?.$ngOptionLabel.toString().trim());
    vendorUser.phone = vendorUser.phone.e164Number;
    // get birthdate with moment
    vendorUser.birthDate = `${new Date(vendorUser.birthdate).getFullYear()}-${new Date(vendorUser.birthdate).getMonth() + 1}-${new Date(vendorUser.birthdate).getDate()}`;
    vendorUser.ipAddress = this.ipUser;


    let branches: any = [];
    vendorUser.branches.map((item: any)=>branches.push({idBranch:item,idProfileVendorUser: vendorUser.idProfileVendorUser}));

    let newUser = {
      "firstName": vendorUser.firstName,
      "lastName": vendorUser.lastName,
      "ipAddress": vendorUser.ipAddress,
      "phone": vendorUser.phone,
      "email": vendorUser.email,
      "identification": vendorUser.identification,
      "gender": vendorUser.gender,
      "birthDate": vendorUser.birthDate,
      "password": "12",
      "stores": [
        {
          "idStore":this.userState.storesPermissions[0].idStore,
          "idProfileVendorUser": vendorUser.idProfileVendorUser,
          "branches": branches
        }
      ]
    };
    this._restUser.postVendorUser(newUser).subscribe({
      next: (response) => {
        //console.log(response);
        let {data} = plainToInstance(ResponseRequest, response);
        this.responseUser = data.user;
        this.password = vendorUser.password;
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'Usuario creado correctamente';
        this._spinner.hide();
      },
      error: err => {
        this._spinner.hide();
        console.log(err);
      }
    });
  }


  public override actionUpdate() {
    super.actionSave();
    this._spinner.show();
    const vendorUser = this.formGroup.getRawValue();
    vendorUser.gender = this.loadGender(vendorUser.gender?.$ngOptionLabel?.toString().trim());
    vendorUser.phone = vendorUser.phone.e164Number;
    // get birthdate with moment
    vendorUser.birthDate = `${new Date(vendorUser.birthdate).getFullYear()}-${new Date(vendorUser.birthdate).getMonth() + 1}-${new Date(vendorUser.birthdate).getDate()}`;
    vendorUser.ipAddress = this.ipUser;


    let branches: any = [];
    vendorUser.branches.map((item: any)=>branches.push({idBranch:item,idProfileVendorUser: vendorUser.idProfileVendorUser}));

    let newUser = {
      "firstName": vendorUser.firstName,
      "lastName": vendorUser.lastName,
      "ipAddress": vendorUser.ipAddress,
      "phone": vendorUser.phone,
      "email": vendorUser.email,
      "identification": vendorUser.identification,
      "gender": vendorUser.gender,
      "birthDate": vendorUser.birthDate,
      "stores": [
        {
          "idStore": this.userState.storesPermissions[0].idStore,
          "idProfileVendorUser": vendorUser.idProfileVendorUser,
          "branches": branches
        }
      ]
    };
    this._restUser.putVendorUser(this.idElement, newUser).subscribe({
      next: (response) => {
        //console.log(response);
        let {data} = plainToInstance(ResponseRequest, response);
        this.responseUser = data.user;
        this.password = vendorUser.password;
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'Usuario actualizado correctamente';
        this._spinner.hide();
      },
      error: err => {
        console.log(err);
        this._spinner.hide();
      }
    });
  }

  public override actionOkDialogSuccess(data?: any) {
    super.actionOkDialogSuccess(data);
    this.isShowSuccessDialog = false;
    void this._router.navigate(['home/vendor/management/access/users']);
  }

  public override actionCancel() {
    super.actionCancel();
    this._router.navigate(['home/vendor/management/access/users']);
  }

  public handleClosePopup(){
    this.isShowSuccessDialog = false;
    void this._router.navigate(['home/vendor/management/access/users']);
  }

}
