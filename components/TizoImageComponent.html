import {Component, EventEmitter, OnInit, Output} from '@angular/core';

@Component({
  selector: 'app-tizo-image',
  templateUrl: './tizo-image.component.html',
  styleUrls: ['./tizo-image.component.scss']
})
export class TizoImageComponent implements OnInit {

  public image: any;
  public imageUrl: any;
  public file: File;
  public listCroppedImages: string[] = [];
  @Output() public listCroppedImagesChange = new EventEmitter<{name:string, base64: string}[]>();

  constructor() { }

  ngOnInit(): void {
  }

  public imagesModelChange($event: any) {
    this.file = $event.target.files[0];
    const reader = new FileReader();
    reader.onload = (e: any) => {
      const name = this.file.name;
      const extension = name.split('.').pop();
      this.imageUrl = {
        name: name,
        extension: extension,
        url: e.target.result,
        path: '',
        description: '',
        base64: e.target.result,
      };
      this.listCroppedImages.push(this.imageUrl.base64);
      let listCroppedImages = this.listCroppedImages.map((item, index) => {
        return {
          name: `product-${index}`,
          base64: item
        };
      });
      this.listCroppedImagesChange.emit(listCroppedImages);
    };
    reader.readAsDataURL(this.file);
  }

  public deleteImage(index: number) {
    this.listCroppedImages.splice(index, 1);
    let listCroppedImages = this.listCroppedImages.map((item, index) => {
      return {
        name: `product-${index}`,
        base64: item
      };
    });
    this.listCroppedImagesChange.emit(listCroppedImages);
  }



}
