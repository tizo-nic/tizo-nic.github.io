import {Component, OnInit} from '@angular/core';
import {ActivatedRoute, Router} from "@angular/router";
import {NgxSpinnerService} from "ngx-spinner";
import {BranchService} from "../../state/branch.service";
import {TizoListComponent} from "../../../../schemes/abstracts/tizo-list.component";
import {StoreApiEntity} from "../../entities/store-api.entity";
import {StoreApiService} from "../../services/store-api.service";
import {StoreApiModel} from "../../models/store-api.model";
import {IBranchModel} from "../../models/branch.model";
import {plainToInstance} from "class-transformer";
import {HttpResponse} from "@angular/common/http";
import {ResponseRequest} from "../../../../schemes/entities/response/response-request";

@Component({
  selector: 'app-api-integrations',
  templateUrl: './api-integrations.component.html',
  styleUrls: ['./api-integrations.component.scss']
})
export class ApiIntegrationsComponent extends TizoListComponent<StoreApiEntity> implements OnInit {

  public storeApiCredentials: StoreApiModel;
  public idStore: number;
  isEmpty: boolean;
  message: string;

  constructor(
    protected override readonly _router: Router,
    private _activatedRoute: ActivatedRoute,
    private _spinner: NgxSpinnerService,
    private _storeApiService: StoreApiService,
    private _branchStore: BranchService
  ) {
    super(_router);
  }

  ngOnInit(): void {
    this.title = 'Api Integrations';
    this._branchStore.branchBS$.subscribe({
      next: (store: IBranchModel) => {
        this.idStore = store.idStore;
        if (this.idStore) {
          this.loadData();
        }
      },
      error: (error) => {
        if (error.status === 400) {
          this.storeApiCredentials = new StoreApiModel();
        }
      }
    });
  }

  public override loadData() {
    this._spinner.show();
    this._storeApiService.getStoreApi(this.idStore).subscribe({
      next: (response: HttpResponse<any>) => {
        let { data} = plainToInstance(ResponseRequest, response);
        this.storeApiCredentials = data.storesApi;
        this._spinner.hide();
      },
      error: (error) => {
        console.log(error);
        this._spinner.hide();
        this.isEmpty = true;
        this.message = 'No hay credenciales de API para esta tienda.';
      }
    });
  }

  actionOk() {
    this.isEmpty = false;
  }
}
