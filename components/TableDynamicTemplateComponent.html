import {
  Component,
  EventEmitter,
  Input,
  Output,
  ViewEncapsulation
} from '@angular/core';

import {TableTemplate} from './table-template';

@Component({
  selector: 'app-dynamic-table',
  template: `<h1>tizo</h1>`,
  encapsulation: ViewEncapsulation.None
})
export class TableDynamicTemplateComponent {

  public isShowWarningDialog: boolean | any;
  public warningDialogMessage: string | any;

  @Input()
  public title: string | any;

  @Input()
  public status: number | any;

  @Input()
  public elements: any[] | any;

  @Input()
  public templates: TableTemplate[] | any;

  @Input()
  public customizeExcelCell: Function| any;

  @Input()
  public showCheckBoxesMode: boolean | any;

  @Input()
  public hasButtonMap: boolean| any;

  @Input()
  public hasButtonView: boolean | any;

  @Input()
  public hasButtonViewDelivery: boolean | any;

  @Input()
  public hasButtonViewDriver: boolean | any;

  @Input()
  public hasButtonUpload: boolean | any;

  @Input()
  public hasButtonEdit: boolean | any;

  @Input()
  public hasButtonRemove: boolean | any;

  @Input()
  public hasButtonClean: boolean | any;

  @Input()
  public hasButtonRestore: boolean | any;

  /**
   * onContentReady
   * */
  @Output()
  public eventEmitterOnContentReady: EventEmitter<any>;

  /**
   * onCellPrepared
   * */
  @Output()
  public eventEmitterOnCellPrepared: EventEmitter<any>;

  /**
   * onToolbarPreparing
   * */
  @Output()
  public eventEmitterOnToolbarPreparing: EventEmitter<any>;

  /**
   * actionOnSelectedRows
   * */
  @Output()
  public eventEmitterOnSelectionChanged: EventEmitter<any>;

  /**
   * onSelectionRowChanged
   * */
  @Output()
  public eventEmitterOnSelectionRowChanged: EventEmitter<any>;

  /**
   * actionMap
   * */
  @Output()
  public eventEmitterActionMap: EventEmitter<any>;

  /**
   * actionView
   * */
  @Output()
  public eventEmitterActionView: EventEmitter<any>;

  /**
   * actionViewDriver
   * */
  @Output()
  public eventEmitterActionViewDriver: EventEmitter<any>;

  /**
   * actionViewDelivery
   * */
  @Output()
  public eventEmitterActionViewDelivery: EventEmitter<any>;

  /**
   * actionUpload
   * */
  @Output()
  public eventEmitterActionUpload: EventEmitter<any>;

  /**
   * actionEdit
   * */
  @Output()
  public eventEmitterActionEdit: EventEmitter<any>;

  /**
   * actionRemove
   * */
  @Output()
  public eventEmitterActionRemove: EventEmitter<any>;

  /**
   * actionClean
   * */
  @Output()
  public eventEmitterActionClean: EventEmitter<any>;

  /**
   * actionRestore
   * */
  @Output()
  public eventEmitterActionRestore: EventEmitter<any>;

  public constants: any;

  constructor() {
    this.eventEmitterOnContentReady = new EventEmitter();
    this.eventEmitterOnCellPrepared = new EventEmitter();
    this.eventEmitterOnToolbarPreparing = new EventEmitter();
    this.eventEmitterOnSelectionChanged = new EventEmitter();
    this.eventEmitterOnSelectionRowChanged = new EventEmitter();
    this.eventEmitterActionMap = new EventEmitter();
    this.eventEmitterActionView = new EventEmitter();
    this.eventEmitterActionViewDriver = new EventEmitter();
    this.eventEmitterActionViewDelivery = new EventEmitter();
    this.eventEmitterActionUpload = new EventEmitter();
    this.eventEmitterActionEdit = new EventEmitter();
    this.eventEmitterActionClean = new EventEmitter();
    this.eventEmitterActionRemove = new EventEmitter();
    this.eventEmitterActionRestore = new EventEmitter();
    this.eventEmitterActionUpload = new EventEmitter();

    this.isShowWarningDialog = false;
  }

  public actionOnCellPrepared(event: any): void {
    this.eventEmitterOnCellPrepared.emit(event);
  }

  public actionOnToolbarPreparing(event: any): void {
    this.eventEmitterOnToolbarPreparing.emit(event);
  }

  public actionOnSelectedRows(event: any): void {
    this.eventEmitterOnSelectionChanged.emit(event);
  }

  public actionSelectedRow(event: any): void {
    this.eventEmitterOnSelectionRowChanged.emit(event.key);
  }

  public actionOnContentReady = (event: any) => {
    this.eventEmitterOnContentReady.emit(event);
  };

  public actionView(data: any): void {
    this.eventEmitterActionView.emit(data);
  }

  public actionViewDriver(data: any): void {
    this.eventEmitterActionViewDriver.emit(data);
  }

  public actionViewDelivery(data: any): void {
    this.eventEmitterActionViewDelivery.emit(data);
  }

  public actionMap(data: any): void {
    this.eventEmitterActionMap.emit(data);
  }

  public actionUpload(event: any, data: any): void {
    this.eventEmitterActionUpload.emit(event);
  }

  public actionEdit(data: any): void {
    this.eventEmitterActionEdit.emit(data);
  }

  public actionClean(data: any): void {
    this.eventEmitterActionClean.emit(data);
  }

  public actionRemove(data: any): void {
    this.eventEmitterActionRemove.emit(data);
  }

  public actionRestore(data: any): void {
    this.eventEmitterActionRestore.emit(data);
  }

}
