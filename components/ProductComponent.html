import { Component, OnInit } from '@angular/core';
import {TizoEditorComponent} from "../../../../schemes/abstracts/tizo-editor.component";

  constructor(
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _router: Router,
    private _activatedRoute: ActivatedRoute,
    private _productRest: ProductService,
    private _restUser: UserService,
    private _branchStore : BranchService,
    private _stateStore: StoreService,
    public _typeProductService: TypeProductService,
  ) {
    super(_spinner, _formBuilder, _router);


    this.formGroup = this._formBuilder.group({
      // Campos de la vista
      isOptionsEnabled: new FormControl(false),
      options: this._formBuilder.array([]),
      wholesalePricesInputs: this._formBuilder.array([]),
      isEnabledWarranty: new FormControl(false),
      isOfferPrice: new FormControl(false),
      isWholesalePrice: new FormControl(false),



      // campos propios del endpoint
      name: new FormControl('', [Validators.required]),
      shortDescription: new FormControl('', Validators.required),
      idProductCategory: new FormControl('', [Validators.required]),
      brand: new FormControl('', [Validators.required]),
      model: new FormControl('', [Validators.required]),
      warranty: new FormControl(1, [Validators.required]),
      cost: new FormControl(1, [Validators.required]),
      quality: [0, [Validators.required]],
      offerPrice: new FormControl(1, [Validators.required]),
      wholesalePrices: new FormControl(1, [Validators.required]),
      price: new FormControl(1, [Validators.required]),
      additionalInformation: new FormControl('', []),

      inventory: this._formBuilder.array([]),
    });
    this.editor = new Editor();
    this.editor.commands.exec();

  }

  public override loadData(){
    forkJoin([
      this._restUser.getUserData(),
      this._productRest.getAllCategories(),
      this._productRest.getProductOptionsList(),
    ]).subscribe({
      next: ([user,categories, options]) => {

        const {data} = plainToInstance(ResponseRequest, user);
        this.listStoreBranches = data.user.storeBranchesPermissions;
        this.listStoreBranches.forEach(() => this.addQuantity());

        const dataCategories = plainToInstance(ResponseRequest, categories);
        this.listCategoriesOptions.next(dataCategories.data.categories);

        // load data for the options select
        const {data: dataOptions} = plainToInstance(ResponseRequest, options);
        this.itemsOptions = dataOptions.options.map((item: string, index: number)=>{
          return {
            id: index,
            name: item
          };
        });

      }
    });

    //Obtener id de la sucursal
    this._branchStore.branchBS$.subscribe({
      next: ({idStore, idStoreBranch})=>{
        this.idStore = idStore;
        this.idBranch = idStoreBranch;
      }
    });
  }

  public ngOnInit() {

    this.entity = new ProductEntity();
    this.loadData();

    this.idElement = this._activatedRoute.snapshot.paramMap.get('idProduct') as string || '';
    if(this.idElement){
      this.initializeFormEdit();
    }else{
      this.options().push(this.newOptions());
      this.variantOptions(0).push(this.newVariant());
      //add wholesale price
      this.wholesalePricesInputs().push(this.newWholesalePricesInputs());
    }
  }


  public initializeFormEdit(): void {

    combineLatest([
      this._productRest.getProductById(this.idElement),
      this.listCategoriesOptions.asObservable(),
      this._productRest.getWholeSalesPrice(this.idElement)
    ]).subscribe({
      next: ([product, categories, wholeSalePrice]) => {

        const {data} = plainToInstance(ResponseRequest, product);
        this.entity = data.product;


        let inventory = data.product.inventories?.map((item: IInventory) => {
          return {
            quantity: item.quantity,
            codeReference: item.codeReference
          };
        });


        let wholesalePrices = data.product.wholesalePrices?.map((item:any) => {
          this.wholesalePricesInputs().push(this.newWholesalePricesInputs());
          return {
            minQuantity: item.minQuantity,
            price: item.price
          };
        });

        let options = data.product.options?.map((item: IOption, index: number) => {
          this.options().push(this.newOptions());
          let variants = item.values?.map((variant: string) => {
            this.variantOptions(index).push(this.newVariant());
            return {
              value: variant
            };
          });
          return {
            optionVariant: item.typeOption,
            variants
          };
        });

        this.isOptionsEnabled = data.product.options.length > 0;
        this.pathCategory = categories?.filter((item: any)=> this.entity.idProductCategory === item.idProductCategory)[0].path;

        // get value of listCategoriesOptions
        this.formGroup.patchValue({
          isOptionsEnabled: data.product.options.length > 0,
          options,
          wholesalePricesInputs: wholesalePrices,
          isEnabledWarranty: true,
          isOfferPrice: true,
          isWholesalePrice: wholesalePrices.length > 0,
          inventory: inventory,


          // campos propios del endpoint
          name: this.entity.name,
          shortDescription: this.entity.shortDescription,
          idProductCategory: this.entity.idProductCategory,
          brand: this.entity.brand,
          model: this.entity.model,
          warranty: this.entity.warranty,
          cost: this.entity.cost,
          quality: this.entity.quality,
          offerPrice: this.entity.offerPrice,
          wholesalePrices: this.entity.wholesalePrice,
          price: this.entity.price,
          additionalInformation: this.entity.additionalInformation,
        });
      },complete: ()=>{
        console.log('complete');
      }
    });

  }

  public override actionSave() {
    let {idProductCategory, name, shortDescription,
      brand, model, warranty, quality,
      options, isOfferPrice, price, offerPrice, cost,
      wholesalePricesInputs, inventory, additionalInformation
    } = this.formGroup.getRawValue();

    // Precio por mayor
    let wholesalePrices = wholesalePricesInputs.map((item: IWholesalePrice) => {
      return {
        "price": item.price,
        "offerPrice": item.price,
        "isOfferPrice": item.minQuantity > 1,
        "minQuantity": item.minQuantity,
      };
    });

    let branchesInventory = inventory.map((item: IInventory, i: number) => {
      return {
        "idBranch": this.listStoreBranches[i].idStoreBranch,
        "codeReference": item.codeReference,
        "quantity": item.quantity,
        "price": price,
        "cost": cost,
        "offerPrice": offerPrice,
        "isOfferPrice": isOfferPrice,
        "warranty": warranty,
        "isInPromotion": false,
      };
    });

    let OptionsVariant = options.map((item: any, index: number) => {
      return {
        "typeOption": item.optionVariant,
        "values": item.variants.map((item: any) => item.value)

  public override actionUpdate() {
    let {idProductCategory, name, shortDescription,
      brand, model, warranty, quality,
      options, isOfferPrice, price, offerPrice, cost,
      wholesalePricesInputs, inventory,
    } = this.formGroup.getRawValue();

    console.log(this.entity);

    // Precio por mayor
    let wholesalePrices = wholesalePricesInputs.map((item: IWholesalePrice, index: number) => {
      return {
        "price": item.price,
        "offerPrice": item.price,
        "isOfferPrice": item.minQuantity > 1,
        "minQuantity": item.minQuantity,
        "idProductWholesalePriceGeneral": this.entity.wholesalePrices[index].idProductWholesalePrice
      };
    });

    let branchesInventory = inventory.map((item: IInventory, i: number) => {
      return {
        "idBranch": this.listStoreBranches[i].idStoreBranch,
        "codeReference": item.codeReference,
        "quantity": item.quantity,
        "price": price,
        "cost": cost,
        "offerPrice": offerPrice,
        "isOfferPrice": isOfferPrice,
        "warranty": warranty,
        "isInPromotion": false,
        "idProductInventory": this.entity.inventories[i].idProductInventory
      };
    });

    let OptionsVariant = options.map((item: any, index: number) => {
      return {
        "typeOption": item.optionVariant,
        "values": item.variants.map((item: any) => item.value)
      };
    });


    let updatedProduct = {
      "idProductCategory": idProductCategory,
      "idProductModel": 0,
      "name": name,
      "brand": brand,
      "shortDescription": shortDescription,
      "largeDescription": 'NO',
      "quality": quality,
      "price": price,
      "cost": cost,
      "offerPrice": offerPrice,
      "isOfferPrice": isOfferPrice,
      "warranty": warranty,
      "isInPromotion": false,
      "inventories": branchesInventory,
      "wholesalePrices": wholesalePrices,
      "options": OptionsVariant,
      "images": this.listCroppedImage,
      "videos": ["string"]
    };

    if (this._typeProductService.getProductType() === 'digital') Object.assign(updatedProduct, {filePath: ""});

    this.isShowQuestionDialog = false;
    this._productRest.putProduct(updatedProduct, this.idElement).subscribe({
      next: () => {
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'Producto editado correctamente';
        this._router.navigate(['/home/products']);
      },
      error: (error) => {
        console.log(error);
      }
    });
  }

  public handleOptionsProduct(){
    this.isOptionsEnabled = !this.isOptionsEnabled;
  }

    super.actionCancel();
    this._router.navigate(['/home/products']);
  }
