import { Component, OnInit } from '@angular/core';
import { UserService } from "../../../services/user/user.service";
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from "@angular/forms";
import { TizoEditorComponent } from "../../../schemes/abstracts/tizo-editor.component";
import { ChangePasswordEntity } from "../../../schemes/entities/change-password.entity";
import { NgxSpinnerService } from "ngx-spinner";
import { Router } from "@angular/router";
import { TizoStorageService } from 'src/app/services/tizo-storage.service';

@Component({
  selector: 'app-forgot-password',
  templateUrl: './forgot-password.component.html',
  styleUrls: ['./forgot-password.component.scss']
})
export class ForgotPasswordComponent extends TizoEditorComponent<ChangePasswordEntity> implements OnInit {


  public countryIndex: number = 0;
  public isShowPopupPassword: boolean = false;
  public changePasswordPost: any;
  public changePassword: UntypedFormGroup;

  public countries: any[] = [
    {name: 'Desarrollo', NIO: 'NIC', img: 'assets/img/countries/tools-line.svg', environment: 'dev'},
    {name: 'Nicaragua', NIO: 'NIC', img: 'assets/img/countries/Nicaragua.svg' , environment: 'ni'},
    //{name: 'Panam', NIO: 'PAN', img: 'assets/img/countries/Panama.svg', environment: 'pa'},
  ];

  constructor(
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _router: Router,
    private _tizoStorageService: TizoStorageService,
    public _userService: UserService
  ) {
    super(_spinner, _formBuilder, _router);
  }

  public ngOnInit(): void {
    this._tizoStorageService.setDevEnvironmentNic(1);

    switch (this._tizoStorageService.getDevEnvironmentNic()) {
    case 'dev':
      this.countryIndex = 0;
      break;
    case 'ni':
      this.countryIndex = 1;
      break;
    case 'pa':
      this.countryIndex = 2;
      break;
    default:
      this.countryIndex = 0;
    }

    this.changePassword = new UntypedFormGroup({
      wordToMatch: new UntypedFormControl('', Validators.required),
      method: new UntypedFormControl('email', Validators.required)
    });

  }

  public changeCountry = (country: number) => {
    this.countryIndex = country;
    this._tizoStorageService.setDevEnvironmentNic(this.countryIndex);
  };

  public onChangePassword(){
    this.isShowPopupPassword = true;
  }

  public onReturn() {
    this._router.navigate(['/signin']);
  }

  public onSubmit(){
    this.changePasswordPost = {
      'wordToMatch': this.changePassword.get('wordToMatch')?.value,
      'method': this.changePassword.get('method')?.value,
      'typeUser': 'vendor'
    };

    this._userService.postRecoveryPassword(this.changePasswordPost).subscribe({
      next: (data) => {
        this.messageDialogSuccess  = 'Se ha enviado las instrucciones para recuperar la contrasea al mtodo seleccionado';
      },
      error: (err) => {
        this.messageDialogSuccess = 'Usuario no encontrado';
      }
    });
  }
}
