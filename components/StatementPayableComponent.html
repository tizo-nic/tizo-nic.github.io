import { Component, OnInit } from '@angular/core';
import {TizoViewComponent} from "../../../../../schemes/abstracts/tizo-view.component";
import {StatementReceivableExtendEntity} from "../../../entities/statement-receivable-extend.entity";
import {ActivatedRoute, Router} from "@angular/router";
import {NgxSpinnerService} from "ngx-spinner";
import {FinanceRestService} from "../../../services/finance-rest.service";
import {BranchService} from "../../../state/branch.service";
import {HttpResponse} from "@angular/common/http";
import {plainToInstance} from "class-transformer";
import {ResponseRequest} from "../../../../../schemes/entities/response/response-request";

@Component({
  selector: 'app-statement-payable',
  templateUrl: './statement-payable.component.html',
  styleUrls: ['./statement-payable.component.scss']
})
export class StatementPayableComponent extends TizoViewComponent<StatementReceivableExtendEntity> implements OnInit {

  idVendor: string;
  idStatement: string;

  constructor(
    protected override readonly _router: Router,
    private _activatedRoute: ActivatedRoute,
    private _spinner: NgxSpinnerService,
    private _financeRestService: FinanceRestService,
    private _branchStore: BranchService
  ) {
    super(_router);
  }
  ngOnInit(): void {
    this.idVendor = this._activatedRoute.snapshot.paramMap.get('idVendor') as string;
    this.idStatement = this._activatedRoute.snapshot.paramMap.get('idStatement') as string;
    this.loadData();
  }
  override loadData() {
    this._spinner.show();
    this.entity = [];
    this._financeRestService.getStatementReceivableDetail(this.idStatement).subscribe({
      next: (response: HttpResponse<any>) => {
        let {data} = plainToInstance(ResponseRequest, response);
        this.entity = plainToInstance(StatementReceivableExtendEntity, data?.statement);
        this._spinner.hide();
        //console.log(this.entity);
      }
    });
  }

}
