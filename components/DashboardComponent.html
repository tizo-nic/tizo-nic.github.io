import { Component, OnInit } from '@angular/core';

// Import components
import {TizoViewComponent} from '../../../schemes/abstracts/tizo-view.component';

// Import router
import {Router} from '@angular/router';
import {BranchService} from "../../vendor-manager/state/branch.service";
import {DashboardRestService} from "../../vendor-manager/services/dashboard-rest.service";
import {IDashboardStatsModel} from "../../vendor-manager/models/dashboard-stats.model";
import {ResponseRequest} from "../../../schemes/entities/response/response-request";
import {plainToInstance} from "class-transformer";
import {NgxSpinnerService} from "ngx-spinner";
import {IBranchModel} from "../../vendor-manager/models/branch.model";
import {UserService} from "../../../services/user/user.service";
import { FormControl, Validators } from "@angular/forms";
import { StoreService } from '../../vendor-manager/state/store.service';
import {TizoStorageService} from '../../../services/tizo-storage.service';

@Component({
  selector: 'app-dashboard',
  templateUrl: './dashboard.component.html',
  styleUrls: ['./dashboard.component.scss']
})
export class DashboardComponent extends TizoViewComponent<any> implements OnInit {

  public greeting: string;
  public idStore: number;
  public idBranch: number;
  public idVendorUser: any;
  public showAcceptTerms: boolean = false;

  public data: IDashboardStatsModel;
  public generalDashboard: IDashboardStatsModel;
  public firstShow: boolean = false;
  public stores: any[];
  public indexStore: number = 0;

  constructor(
    protected override _router: Router,
    private _branchStore : BranchService,
    private _dashboardRestService: DashboardRestService,
    private _spinner: NgxSpinnerService,
    private _userRest: UserService,
    private _storeService: StoreService,
    private _tizoStorage: TizoStorageService
  ) {
    super(_router);
  }

  ngOnInit(): void {

    this._storeService.indexStoreSelected$.subscribe({
      next: (store: any) => {
        this.indexStore = store;
      }
    });


    //si es true mostrar si es false no mostrar
    this.showAcceptTerms = this._tizoStorage.isAcceptedTerms() !== 'true';

    //si ya se ha aceptado los trminos no mostrar el modal
    this.showAcceptTerms ? this._tizoStorage.setFirstShow(false) : null;

    //identificamos si es la primera vez que se muestra al entrar a la pagina
    this._tizoStorage.needToShow$.subscribe((needToShow: any) => {
      this.firstShow = needToShow;
    });


    this.greeting = new Date().getHours() < 12 ? 'Buenos das!' : new Date().getHours() < 18 ? 'Buenas tardes!' : 'Buenas noches!';

    this._userRest.getStores().subscribe({
      next: (response: any) => {
        this.stores = response.data.stores;
        // this.idStore = this.stores[this.indexStore].branches[0].idStore;
        // this.idBranch = this.stores[this.indexStore].branches[0].idStoreBranch;
        // if (this.idBranch) {
        //   this.loadData(this.idBranch);
        // }
      }
    });

    this._userRest.getUserData().subscribe({
      next: (response: any) => {
        this.idVendorUser = response.data.user.idVendorUser;
      }
    });

    this._branchStore.branchBS$.subscribe({
      next: (store: IBranchModel ) => {
        this.idBranch = store.idStoreBranch;
        this.idStore = store.idStore;
        // console.log(this.idBranch);
        // console.log(this.idStore);
        if (this.idBranch) {
          this.loadData(this.idBranch);
        }
      },
      error: (error) => {
        console.log(error);
      },
    });
  }

  public selectStore(index: number){
    this._storeService.setIndexStoreSelected(index);
    this.indexStore = index;
  }

  public navigateToTotalDeliveries(){
    this._router.navigate([`/home/vendor/${this.idVendorUser}/deliveries`]);
  }

  public navigateToTotalDeliveriesAPI(){
    this._router.navigate([`/home/vendor/${this.idVendorUser}/deliveries-api`]);
  }

  public navigateToAccountsPayble(){
    this._router.navigate([`/home/vendor/${this.idVendorUser}/finance/accounts-payable`]);
  }

  public actionAcceptTerms(): void {
    this._tizoStorage.setAcceptedTerms('true');
    this.showAcceptTerms = false;
  }

  public onHidden(){
    this._tizoStorage.setFirstShow(false);
  }

  public override loadData(idBranch: any) {
    this._spinner.show();
    let query = idBranch ? `?idBranch=${idBranch}`: '';
    this._dashboardRestService.getDashboard(this.idStore, query).subscribe({
      next: (response: any) => {
        let {data} = plainToInstance(ResponseRequest, response);
        this.data = data;

        //controlando si no hay cuentas por pagar o no hay cuentas por cobrar poner 0 en el html
        ((this.data.accountsPayable !== undefined) ? (this.data.accountsPayable = this.data.accountsPayable) : this.data.accountsPayable = 0);
        ((this.data.accountsReceivable !== undefined) ? (this.data.accountsReceivable = this.data.accountsReceivable) : this.data.accountsReceivable = 0);

        this._spinner.hide();
      },
      error: (error: any) => {
        this._spinner.hide();
      }
    });

    this._dashboardRestService.getDashboardGeneral(this.idStore).subscribe({
      next: (response: any) => {
        let generalDashboardInfo = plainToInstance(ResponseRequest, response);
        this.generalDashboard = generalDashboardInfo.data;

        //controlando si no hay cuentas por pagar o no hay cuentas por cobrar poner 0 en el html
        ((this.generalDashboard.accountsPayable !== undefined) ? (this.generalDashboard.accountsPayable = this.generalDashboard.accountsPayable) : this.generalDashboard.accountsPayable = 0);
        ((this.generalDashboard.accountsReceivable !== undefined) ? (this.generalDashboard.accountsReceivable = this.generalDashboard.accountsReceivable) : this.generalDashboard.accountsReceivable = 0);

        this._spinner.hide();
      },
      error: (error: any) => {
        this._spinner.hide();
      }
    });
  }

}
