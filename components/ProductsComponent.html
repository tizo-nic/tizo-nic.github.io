import { Component, OnInit } from '@angular/core';
import {ApiTestService} from "../../services/api-test.service";
import {Router} from "@angular/router";
import {ProductEntity} from "../../entities/product.entity";
import {TizoListComponent} from "../../../../schemes/abstracts/tizo-list.component";
import {ProductService} from "../../services/product.service";
import {NgxSpinnerService} from "ngx-spinner";
import {TypeProductService} from "../../services/type-product.service";
import {forkJoin} from "rxjs";
import {plainToInstance} from "class-transformer";
import {ResponseRequest} from "../../../../schemes/entities/response/response-request";

@Component({
  selector: 'app-products',
  templateUrl: './products.component.html',
  styleUrls: ['./products.component.scss']
})
export class ProductsComponent extends TizoListComponent<ProductEntity> implements OnInit {


  public showModalNewProduct: boolean;
  public stats:{
    totalProducts: number,
    totalProductsWithoutDescription: number,
    totalProductsWithoutImage: number,
    totalProductsWithoutVideos: number,
  } = {
      totalProducts: 0,
      totalProductsWithoutDescription: 0,
      totalProductsWithoutImage: 0,
      totalProductsWithoutVideos: 0,
    };

  // variables para la vista
  public override isShowQuestionDialog: boolean = false;
  public override messageDialogQuestion: string = 'Est seguro de eliminar el producto?';

  public override isShowSuccessDialog: boolean = false;
  constructor(
    protected override readonly _router: Router,
    private _spinner: NgxSpinnerService,
    private _apiTestService: ApiTestService,
    private _productRest: ProductService,
    public _typeProductService: TypeProductService,
  ) {
    super(_router);
  }

  public override messageDialogSuccess: string = 'El producto se ha eliminado correctamente';

  ngOnInit(): void {
    this.entity = new ProductEntity();
    this.loadData();
  }

  public override loadData(): void {
    this._spinner.show();
    this.templates = new ProductEntity().setTableTemplate('simple');

    forkJoin([
      this._productRest.getProducts(),
      this._productRest.getStatsProduct()
    ]).subscribe({
      next: ([responseProducts, responseStats]) => {

        let {data: dataProducts}  = plainToInstance(ResponseRequest, responseProducts);

        this.entities = dataProducts.products;
        this.entities = this.entities.filter((product: any) => product.statusName === 'Creado');


        let {data: dataStats}  = plainToInstance(ResponseRequest, responseStats);
        this.stats = dataStats.stats;

        this._spinner.hide();
      },
      error: (error) => {
        this._spinner.hide();
      }
    });
  }

  public override actionRemove(data: ProductEntity) {
    this.entity = data;
    this.isShowQuestionDialog = true;
  }

  public handleClickConfirmDelete() {
    this._productRest.deleteProduct(this.entity.idProduct).subscribe({
      next: (data: any) => {
        this.isShowQuestionDialog = false;
        this.isShowSuccessDialog = true;
        this.loadData();
      }
    });
  }

  public override actionEdit(data: ProductEntity) {
    super.actionEdit(data);
    this._router.navigate(['/home/product', data.idProduct]);
  }

  public handleProductDigital() {
    this._typeProductService.setProductType('digital');
    this._router.navigate(['/home/product']);
  }

  public handleProductPhysical() {
    this._typeProductService.setProductType('physical');
    this._router.navigate(['/home/product']);
  }

}
