import { Component, OnInit, ViewChild } from "@angular/core";
import { Router } from "@angular/router";
import { TizoViewComponent } from "src/app/schemes/abstracts/tizo-view.component";
import { CustomerService } from "../customer.service";
import {UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators} from "@angular/forms";
import {CountryISO} from "ngx-intl-tel-input";
import { ActivatedRoute } from "@angular/router";

import {
  CdkDropList,
  CdkDropListGroup,
} from "@angular/cdk/drag-drop";
import { ViewportRuler } from "@angular/cdk/overlay";
import { WaypointService } from "../../logistic/state/Observables/waypoint.service";
import { WaypointModel } from "../../logistic/models/delivery/waypoint.model";
import {PositionCustomerService} from "../../logistic/state/Observables/position-customer.service";
import { CustomerEntity } from "../customer.entity";
import { plainToInstance } from "class-transformer";
import { ResponseRequest } from "src/app/schemes/entities/response/response-request";

declare const google: any;

@Component({
  selector: 'app-customer',
  templateUrl: './customer.component.html',
  styleUrls: ['./customer.component.scss']
})
export class CustomerComponent extends TizoViewComponent<any> implements OnInit {

  zoom: number;
  public latitude: number;
  public longitude: number;
  public addressGeo: string;
  public addressStreet: string;
  public city: string;
  public state: string;
  public country: string;

  public idCustomer: string;
  public isSuccessVisible = false;
  public successMessage: string;

  public formGroup: UntypedFormGroup;
  public selectedCountryISO: CountryISO;
  public preferredCountries: string[] = [];
  public idWayPoint: number;
  public idVendor: string;

  public pointMap: any;

  public addressHuman: string;
  public instructions: string;

  public items: WaypointModel[] = [];
  public stepStatus: number = 1; // 1 - Lista de direcciones, 2 - Agregar direcciones, 3 - Agregar paquetes, 4 - Resumen del envio

  // Dialogs
  public override isShowWarningDialog: boolean = false;
  public override messageDialogWarning: string = "";
  public customerPost: any;


  public firstname: string;
  public lastname: string;
  public phone: any;
  public personIdentification: string;
  public personEmail: string;


  constructor(
    protected override _router: Router,
    private viewportRuler: ViewportRuler,
    private _waypointService: WaypointService,
    private _positionCustomerService: PositionCustomerService,
    private _customerService: CustomerService,
    public _formBuilder: UntypedFormBuilder,
    private _activeRoute: ActivatedRoute
  ) {
    super(_router);
    this.target = null;
    this.source = null;

    this.preferredCountries = [
      CountryISO.Nicaragua,
      CountryISO.Panama,
      CountryISO.Guatemala,
      CountryISO.UnitedStates
    ];

    this.selectedCountryISO = CountryISO.Nicaragua;
  }

  public loadFormBuilder(){

    if(this.idCustomer){
      this._customerService.getCustomerById(this.idElement, this.idCustomer).subscribe({
        next: (response: any) => {
          //console.log(response);
          let {data} = plainToInstance(ResponseRequest, response);
          this.entity = plainToInstance(CustomerEntity, data.customer);

          this.firstname = this.entity.name;
          this.phone = (this.entity.phone.startsWith('+1')) ? this.entity.phone.slice(2) : this.entity.phone.slice(4);
          this.personIdentification = this.entity.identification;
          this.personEmail = this.entity.email;
          this.instructions = this.entity.instructions;
          this.addressHuman = this.entity.addressHuman;
          this.latitude = this.entity.latitude;
          this.longitude = this.entity.longitude;
          this._positionCustomerService.editPosition(this.latitude, this.longitude);
          // this._customerService.setCustomerPoint(this.entity);
          this.addressGeo = this.entity.addressGeo;
          this.addressStreet = this.entity.addressStreet;

          //this._customerService.setCustomerPoint(this.entity);
        },
        error: (err) => {
          console.log("error cargando cliente");
        }
      });

    }

    this.formGroup = this._formBuilder.group({
      firstname: new UntypedFormControl(this.firstname, [Validators.required]),
      lastname: new UntypedFormControl('', [Validators.required, Validators.minLength(6)]),
      phone: new UntypedFormControl('', [Validators.required, Validators.minLength(6)]),
      personEmail: new UntypedFormControl('', [Validators.required, Validators.email]),
      personIdentification: new UntypedFormControl('', [Validators.minLength(6)]),
    });
  }

  ngOnInit(): void {
    this.title = 'Direcciones de envo';
    this.description = '';
    this.target = null;
    this.source = null;

    this.idElement = this._activeRoute.snapshot.paramMap.get('idStore') as string;
    this.idCustomer = this._activeRoute.snapshot.paramMap.get('idCustomer') as string;
    this.idVendor = this._activeRoute.snapshot.paramMap.get('idVendor') as string;

    this.zoom = 11;

    this.entity = new CustomerEntity();

    this.loadFormBuilder();
  }

  public selectPointByMap($event: any) {
    this.pointMap = $event;
  }

  public actionConfirmPersonContact() {

    let waypoint = {
      type: 'DROP_OFF',
      city: this.pointMap.city,
      country: this.pointMap.country,
      state: this.pointMap.state,
      addressStreet: this.pointMap.addressStreet,
      personName: this.firstname,
      personIdentification: this.personIdentification,
      personEmail: this.personEmail,
      phone: this.phone.e164Number,
      addressHuman: this.addressHuman,
      addressGeo: this.pointMap.addressGeo,
      instructions: this.instructions,
      latitude: this.pointMap.latitude,
      longitude: this.pointMap.longitude,
    };

    this.customerPost = {
      "idMunicipality": 1,
      "addressGeo": waypoint.addressGeo,
      "addressHuman": waypoint.addressHuman,
      "city": waypoint.city,
      "latitude": waypoint.latitude,
      "name": waypoint.personName,
      "longitude": waypoint.longitude,
      "instructions": waypoint.instructions,
      "identification": waypoint.personIdentification,
      "phone": waypoint.phone,
      "email": waypoint.personEmail === undefined ? '': waypoint.personEmail,
      "state": waypoint.state,
      "country": waypoint.country,
      "addressStreet": waypoint.addressStreet,
    };

    this._customerService.postCustomer(this.idElement, this.customerPost).subscribe({
      next: (data: any) => {
        this.successMessage = 'Cliente agregado correctamente';
        this.isSuccessVisible = true;
      },
      error: (err) => {
        console.log(err);
        this.messageDialogWarning = 'Debe proporcionar la informacin solicitada del cliente';
        this.isShowDangerDialog = true;
      }
    }
    );
  }

  public actionEditPersonContact() {

    let waypoint = {
      type: 'DROP_OFF',
      city: this.pointMap.city,
      country: this.pointMap.country,
      state: this.pointMap.state,
      addressStreet: this.pointMap.addressStreet,
      personName: this.firstname,
      personIdentification: this.personIdentification,
      personEmail: this.personEmail,
      phone: this.phone.e164Number,
      addressHuman: this.addressHuman,
      addressGeo: this.pointMap.addressGeo,
      instructions: this.instructions,
      latitude: this.pointMap.latitude,
      longitude: this.pointMap.longitude,
    };

    this.customerPost = {
      "idMunicipality": 1,
      "addressGeo": waypoint.addressGeo,
      "addressHuman": waypoint.addressHuman,
      "city": waypoint.city,
      "latitude": waypoint.latitude,
      "name": waypoint.personName,
      "longitude": waypoint.longitude,
      "instructions": waypoint.instructions,
      "identification": waypoint.personIdentification,
      "phone": waypoint.phone,
      "email": waypoint.personEmail,
    };

    this._customerService.putCustomer(this.idElement, this.idCustomer, this.customerPost).subscribe({
      next: (data: any) => {
        this.successMessage = 'Cliente editado correctamente';
        this.isSuccessVisible = true;
      },
      error: (err) => {
        console.log(err);
      }
    }
    );
  }

  public actionCancel() {
    this._router.navigate([`/home/vendor/${this.idVendor}/customers`]);
  }

  public eventEmitterClose(){
    this.isSuccessVisible = false;
    this._router.navigate([`/home/vendor/${this.idVendor}/customers`]);
  }

  public actionOk(){
    this.isSuccessVisible = false;
    this._router.navigate([`/home/vendor/${this.idVendor}/customers`]);
  }

  /*===================================================================*/

  @ViewChild(CdkDropListGroup) listGroup: CdkDropListGroup<CdkDropList>;
  @ViewChild(CdkDropList) placeholder: any;

  public target: any;
  public source: any;

}
