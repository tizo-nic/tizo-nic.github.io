import {AfterViewInit, Component, ElementRef, OnInit, ViewChild} from '@angular/core';
import {NgxSpinnerService} from "ngx-spinner";
import {UserService} from "../services/user/user.service";
import {UserStateService} from '../state/Observables/user-state.service';
import { NotificationsService } from "../services/notifications.service";
import {plainToInstance} from "class-transformer";
import {StoreEntity} from "./vendor-manager/entities/store.entity";
import {StoreRestService} from "./vendor-manager/services/store-rest.service";
import {IStoreModel} from "./vendor-manager/models/store.model";
import {StoreService} from "./vendor-manager/state/store.service";
import {ResponseRequest} from "../schemes/entities/response/response-request";
import {SidebarStyleService} from './logistic/state/Observables/sidebar-style.service';
import {forkJoin, Observable} from "rxjs";
import {AdminSystemService} from "../services/admin-system/admin-system.service";
import {TypeSystem} from "../schemes/enum/type-system";

@Component({
  selector: 'app-pages',
  templateUrl: './pages.component.html',
  styleUrls: ['./pages.component.scss']
})
export class PagesComponent implements OnInit, AfterViewInit{

  @ViewChild('main') elementView: ElementRef;

  public isHideSidebar = false;
  public isHoverSideBar = true;
  public typeUser: string;
  public idUserVendor: number;
  public isVisible$: Observable<boolean>;
  public indexStore: number = 0;
  public typeSystem$: Observable<TypeSystem>;
  public storeEntity: IStoreModel = new StoreEntity();
  public stores: any[] | number[];


  //lógica visual del botón flotante
  public isShowMenuFloat: boolean = false;

  constructor(
    private readonly _spinner: NgxSpinnerService,
    private _restUser: UserService,
    private _userStateService: UserStateService,
    private _restStore: StoreRestService,
    private _stateStore: StoreService,
    private _notificationsService:NotificationsService,
    private _sidebarStyleService: SidebarStyleService,
    private _typeSystemService: AdminSystemService
  ) {
    this.typeSystem$ = this._typeSystemService.getSystem$;
  }

  ngOnInit(): void {
    this._spinner.show();
    this.isVisible$ = this._sidebarStyleService.getIsVisible;
    this.loadData();
  }

  public ngAfterViewInit() {
    this.isHideSidebar = false;
  }

  public loadData(){
    // Obtiene los permisos del usuario
    // ====================================================================//
    this._stateStore.indexStoreSelected$.subscribe({
      next: (index) => {
        this.indexStore = index;
        // Obtiene los permisos del usuario
        forkJoin([
          this._restUser.getStores(),
          this._restUser.getUserData()
        ]).subscribe({
          next: ([ responseStores, responseUser  ])=>{

            let dataStores = plainToInstance(ResponseRequest, responseStores);
            this.stores = dataStores.data.stores;


            /**==========================================*/

            let {data} = plainToInstance(ResponseRequest, responseUser);
            this.storeEntity = plainToInstance(StoreEntity, data.user.storesPermissions[this.indexStore]);
            this._stateStore.addStore(this.storeEntity);
            //====================================================================//

            this._notificationsService.notifyStatusDelivery(data.user.username);
            this._userStateService.setCurrentUser(data);
            this.typeUser = data.type_user;
            // Método para cargar solo configuraciones de usuario courier
            this.idUserVendor = (this.typeUser === 'vendor_user') ? data.user?.idVendorUser : 0;
            void this._spinner.hide();

          },
          error: (err: any) => {
            console.log(err);
            this._spinner.hide();
          }
        });
      }
    });


  }

  public selectStore(index: number){
    this._stateStore.setIndexStoreSelected(index);
    this.indexStore = index;
  }

  public onClickedOutside(e: Event) {
    this.isShowMenuFloat = false;
  }

}
