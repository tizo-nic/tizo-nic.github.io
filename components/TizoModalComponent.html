import {Component, EventEmitter, Input, OnInit, Output} from '@angular/core';
import {animate, style, transition, trigger} from "@angular/animations";

@Component({
  selector: 'app-tizo-modal',
  templateUrl: './tizo-modal.component.html',
  styleUrls: ['./tizo-modal.component.scss'],
  animations: [
    // add effect easing
    trigger('dialog', [
      transition('void => *', [
        style({
          transform: 'scale3d(.3, .3, .3)' },
        ),
        animate(200),
      ]),
      transition('* => void', [
        animate(200, style({ transform: 'scale3d(.0, .0, .0)' })),
      ]),
    ]),
  ],
})
export class TizoModalComponent implements OnInit {

  @Input() modalTitle: string;
  //create input for show modal
  @Input() visible: boolean;

  @Input() showControlOkButton: boolean = false;

  @Input() showControlCancelButton: boolean = false;

  @Input() showModalTittle: boolean = true;

  @Input() shadingBackground: boolean = true;

  @Input() closeOnOutsideClick: boolean = false;

  @Input() width: string;

  @Input() height: string;

  @Output() eventEmitterClose = new EventEmitter<boolean>();

  @Output() eventEmitterOk: EventEmitter<any>;

  constructor() {
    this.eventEmitterOk = new EventEmitter<null>();
  }

  ngOnInit(): void {
  }


  public handleOkModal() {
    this.eventEmitterOk.emit(null);
  }

  public handleCloseModal() {
    this.visible = false;
    this.eventEmitterClose.emit(this.visible);
  }

  public onClickedInside() {
    if(this.closeOnOutsideClick){
      this.visible = false;
      this.eventEmitterClose.emit(this.visible);
    }
  }

  public onClickedOutside(e: Event) {
    this.visible = false;
  }

}
