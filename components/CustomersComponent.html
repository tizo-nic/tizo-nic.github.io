import { Component, OnInit } from '@angular/core';
import {plainToInstance} from 'class-transformer';
import {NgxSpinnerService} from 'ngx-spinner';
import {CustomerService} from "../customer.service";
import {ActivatedRoute, Router} from '@angular/router';
import {CustomerEntity} from "../customer.entity";
import {TizoListComponent} from "../../../schemes/abstracts/tizo-list.component";
import {ResponseRequest} from '../../../schemes/entities/response/response-request';
import {StoreService} from "../../vendor-manager/state/store.service";
import { WaypointService } from '../../logistic/state/Observables/waypoint.service';

@Component({
  selector: 'app-customers',
  templateUrl: './customers.component.html',
  styleUrls: ['./customers.component.scss']
})
export class CustomersComponent extends TizoListComponent<CustomerEntity> implements OnInit {

  private idStore: string;
  private storeEntity: any;
  public wayPoints: any[] = [];

  constructor(
    protected override _router: Router,
    private _activeRoute: ActivatedRoute,
    private _customerService: CustomerService,
    private _spinner: NgxSpinnerService,
    private _stateStore: StoreService,
    private _wayPointStore: WaypointService
  ) {
    super(_router);
  }

  ngOnInit(): void {
    this._stateStore.ztore$.subscribe({
      next: (store) => {
        this.storeEntity = store;
        this.loadData();
        this.loadWaypoints();
      },
      error: (error) => {},
    });

    this.title = 'Lista de clientes de tizo';
    this.idElement = this._activeRoute.snapshot.paramMap.get('idVendor') || '';
    this.loadBreadcrumb();
  }

  public loadBreadcrumb():void{

    this.title = 'Clientes';
    this.description = 'Lista de clientes';

    this.breadcrumbs.push( {
      name: 'Home',
      url: '/home',
      isActive: true
    },{
      name: 'Clientes que realizan pedidos',
      url: `/home/vendor/${this.idElement}/customers`,
      isActive: true
    });
  }

  public override loadData() {
    //void this._spinner.show();
    this.entities = [];
    this.idStore = this.storeEntity.idStore;
    this.templates = new CustomerEntity().setTableTemplate('simple');
    this._customerService.getAllCustomers(this.storeEntity.idStore).subscribe({
      next: (response: any) => {
        //console.log(response);
        let {data} = plainToInstance(ResponseRequest, response);
        data?.customers.map((customer: any) => {
          this.entities.push(plainToInstance(CustomerEntity, customer));
        });
        void this._spinner.hide();
      },
      error: (err) => {
        console.log(err);
        void this._spinner.hide();
      }
    });
  }

  public override actionSave() {
    super.actionSave();
    this._router.navigate([`/home/vendor/${this.idElement}/stores/${this.idStore}/customers`]);
  }

  public loadWaypoints() {
    this._wayPointStore.getWayPoints().subscribe({
      next: (wayPoints) => {
        this.wayPoints = wayPoints;
      }
    });
  }

  public override actionEdit(data: CustomerEntity) {
    super.actionEdit(data);
    this._wayPointStore.setWayPointA({
      type: 'DROP_OFF', // '[PICK_UP | DROP_OFF]';
      city: data.city,
      country: 'Nicaragua',
      state: data.state ? data.state : 'Managua',
      addressStreet: data.addressStreet ? data.addressStreet : '',
      personName: data.name,
      personEmail: data.email,
      personIdentification: data.identification,
      phone: data.phone,
      addressHuman: data.addressHuman,
      addressGeo: data.addressGeo,
      instructions: data.instructions,
      latitude: data.latitude,
      longitude: data.longitude,
    });

    this._router.navigate([`/home/vendor/${this.idElement}/stores/${this.idStore}/customers/${data.idCustomerAddress}`]);
  }

  public override actionShipment(data: CustomerEntity) {
    super.actionShipment(data);

    let lenght = this.wayPoints.length;
    let type = (lenght === 0 ? 'PICK_UP' : 'DROP_OFF');

    let point = {
      type: type, // '[PICK_UP | DROP_OFF]';
      city: data.city,
      country: 'Nicaragua',
      state: data.state ? data.state : 'Managua',
      addressStreet: data.addressStreet ? data.addressStreet : '',
      personName: data.name,
      personEmail: data.email,
      personIdentification: data.identification,
      phone: data.phone,
      addressHuman: data.addressHuman,
      addressGeo: data.addressGeo,
      instructions: data.instructions,
      latitude: data.latitude,
      longitude: data.longitude,
    };

    // agregar waypoint
    lenght === 0 ? this._wayPointStore.setWayPointA(point) : this._wayPointStore.setWayPointB(point);

    this._router.navigate([`/home/vendor/${this.idElement}/create-shipment`]);
  }

  public onCloseQuestionDialog(): void {
    this.isShowQuestionDialog = false;
  }

  public override actionRemove(data: CustomerEntity): void {
    this.entity = data.idCustomerAddress;
    this.isShowQuestionDialog = true;
  }

  public dialogConfirmDelete(){
    this.isShowQuestionDialog = false;
    void this._spinner.show();
    this._customerService.deleteCustomer(this.idStore, this.entity).subscribe({
      next: () => this.loadData(),
      error: (err :any) => {
        console.log(err);
        void this._spinner.hide();
      }
    });
  }

}
