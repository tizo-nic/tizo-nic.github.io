import { HttpHeaders, HttpParams } from '@angular/common/http';
import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { UserService } from 'src/app/services/user/user.service';

@Component({
  selector: 'app-reset-password-forgiven',
  templateUrl: './reset-password-forgiven.component.html',
  styleUrls: ['./reset-password-forgiven.component.scss']
})
export class ResetPasswordForgivenComponent implements OnInit {

  // Create number random in range 1-3
  public randomNumber: Number = Math.floor(Math.random() * 3) + 1;

  token: string;

  constructor(
    private _userService: UserService,
    private _route: ActivatedRoute
  ) { }

  ngOnInit(): void {
    this.token = this._route.snapshot.queryParams['token'];
  }

  resetPassword = new UntypedFormGroup({
    password: new UntypedFormControl('', Validators.required),
    password_again: new UntypedFormControl('', Validators.required)
  });

  onResetPassword(){
    const headers = new HttpHeaders().append('environment', 'dev');
    const password = {'password': 'xd'};
    const params = new HttpParams().append('key', this.token);
    if (this.resetPassword.get('password')?.value === this.resetPassword.get('password_again')?.value){
      this._userService.postRecoveryPassword2(password, headers, params).subscribe({
        next: (data) => {},
        error: (err) => {}
      });
    }
  }

}
