import {Component, NgZone, OnInit, Output, EventEmitter} from '@angular/core';
import {TizoMapEditorComponent} from "../../../../schemes/abstracts/tizo-map-editor.component";
import {BranchEntity} from "../../entities/branch.entity";
import {NgxSpinnerService} from "ngx-spinner";
import {UntypedFormBuilder} from "@angular/forms";
import {MapsAPILoader} from "@agm/core";
import {ActivatedRoute, Router} from "@angular/router";
import {MapService} from "../../../../services/map.service";
import {BranchRestService} from "../../services/branch-rest.service";
import {plainToInstance} from "class-transformer";
import {ResponseRequest} from "../../../../schemes/entities/response/response-request";

@Component({
  selector: 'app-branch',
  templateUrl: './branch.component.html',
  styleUrls: ['./branch.component.scss']
})
export class BranchComponent extends TizoMapEditorComponent<BranchEntity> implements OnInit {


  public idStore: string;
  public idVendor: string;

  public override latitude: number;
  public override longitude: number;
  public selectedPoint: boolean = false;
  public override zoom: number;

  @Output() selectPoint = new EventEmitter<any>();


  public isClosedPolygon: boolean;
  public isDrawInternalPolygon: boolean;

  public areaLatitude: number;
  public areaLongitude: number;

  public schedule: any;

  public override isShowSuccessDialog = false;

  constructor(
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _mapsAPILoader: MapsAPILoader,
    protected override readonly _router: Router,
    protected override readonly _ngZone: NgZone,
    protected override readonly _mapService: MapService,
    private _activatedRoute: ActivatedRoute,
    private _restBranch: BranchRestService,
  ) {
    super(_spinner, _formBuilder, _mapsAPILoader, _router, _ngZone, _mapService);
    this.isClosedPolygon = false;
    this.isDrawInternalPolygon = false;
    this.selectedPoint = false;
    this.zoom = 11;
  }

  ngOnInit(): void {
    this.idElement = this._activatedRoute.snapshot.paramMap.get('idBranch') as string;
    this.idStore = this._activatedRoute.snapshot.paramMap.get('idStore') as string;
    this.idVendor = this._activatedRoute.snapshot.paramMap.get('idVendor') as string;

    this.loadData();
  }

  public override loadData(){
    this.entity = new BranchEntity();
    this.initializeMap();
    this.loadVendorArea();
    if (this.idElement) {
      this.initializeFormEditor();
    }
  }

  public async loadVendorArea(){


    this._mapService.getPosition().then(position => {
      this.areaLongitude = position.lng;
      this.areaLatitude = position.lat;
    });

    this.polygonOptions.editable = false;
    this.polygonOptions.draggable = false;
  }

  public actionChangeSimpleSelector($event: any){
    if($event.field === 'idMunicipality' && $event.selectable){
      this.handleChangeCourierArea($event);
    }
  }

  public handleChangeCourierArea($event: any){
    this.entity.CourierAreaEntity = $event.selectable;
    this.entity.geojson = $event.selectable.geojson;
    this.entity.coordinates = JSON.parse($event.selectable.geojson);
    this.polygonOptions.editable = false;
    this.polygonOptions.clickable = false;
    this.actionClean();
    //this.initializeDrawingManager();
    this.initializeCoordinates();
  }

  public initializeFormEditor() {
    this.selectedPoint = true;
    this._restBranch.getBranchById(this.idStore, this.idElement).subscribe({
      next: (response) => {
        let {data} = plainToInstance(ResponseRequest, response);
        this.entity = plainToInstance(BranchEntity, data.branch);
        this.latitude = this.entity.latitude;
        this.longitude = this.entity.longitude;
        this.getAddress(this.latitude, this.longitude);
      },
      error: (error) => {
      }
    });
  }

  public async getAddress(lat: number, lng: number): Promise<void> {
    let addressStreet = '', address = '', country = '', state = '', city = '';

    let geoCoder = new google.maps.Geocoder();
    let latLng = new google.maps.LatLng(lat, lng);
    const request: google.maps.GeocoderRequest = {
      location: latLng
    };

    await geoCoder.geocode(request, (results: string | any[], status: any) => {
      if (results !== null) {
        address = results[0].formatted_address;
        const length = results.length;
        addressStreet = results[length - 4]?.formatted_address.split(',')[0];
        country = results[length - 1]?.formatted_address;
        state = results[length - 2]?.formatted_address.split(',')[0];
        city = results[length - 3]?.formatted_address.split(',')[0];
        this.addPoints(lat, lng, addressStreet, address, country, state, city);
      }
    });
  }

  public addPoints(lat: number, lng: number,addressStreet: string, address: string, country: string, state: string, city: string){

    // Orden del proceso
    // 1. Agregar el punto A al mapa
    //   a. Verificar si el punto ya existe
    //   b.

    // 2. Agregar el punto B al mapa
    //   a. Verificar si el punto ya existe

    let point = {
      latitude: lat,
      longitude: lng,
      addressGeo: address,
      addressStreet: address,
      city: city,
      state: state,
      country: country
    };

    //this.itemsPoint[0] = point;
    this.selectPoint.emit(point);

    this.latitude = lat;
    this.longitude = lng;

  }

  public override actionUpdate() {
    super.actionCancel();
    let data = this.formGroup.getRawValue();

    if(!this.entity.addressGeo){
      this.isShowWarningDialog = true;
      this.messageDialogWarning = 'Por favor, seleccione un punto dentro del polígono, para ubicar la sucursal.';
      return;
    }

    let updateBranch = {
      "name": data.name,
      "idMunicipality": 1,
      "addressGeo": this.entity.addressGeo,
      "addressHuman": data.addressHuman,
      "latitude": this.latitude,
      "longitude": this.longitude,
    };

    this._restBranch.putBranch(this.idStore, this.idElement, updateBranch).subscribe({
      next: (response) => {
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'La sucursal se ha actualizado correctamente.';
        //this._router.navigate([`/home/courier/${this.idCourier}/coverage/branches`]);
      },
      error: (error) => {},
    });
  }

  public override actionCancel() {
    this._router.navigate([`home/vendor/${this.idVendor}/store/${this.idStore}`]);
  }

  public override actionSave() {
    super.actionSave();


    let data = this.formGroup.getRawValue();

    if(!this.entity.addressGeo){
      this.isShowWarningDialog = true;
      this.messageDialogWarning = 'Por favor, seleccione un punto dentro del polígono, para ubicar la sucursal.';
      return;
    }

    let newBranch = {
      "name": data.name,
      "idMunicipality": 1,
      "addressGeo": this.entity.addressGeo,
      "addressHuman": data.addressHuman,
      "latitude": this.latitude,
      "longitude": this.longitude,
    };

    this._restBranch.postBranch(this.idStore, newBranch).subscribe({
      next: (response) => {
        this.isShowSuccessDialog = true;
        this.messageDialogSuccess = 'Sucursal creada correctamente.';
      },
      error: (error) => {},
    });

  }

  public eventEmitterGetSchedule($event: any){
    this.schedule = $event;
  }

  public actionSuccessOk($event: any){
    this.isShowSuccessDialog = false;
    this._router.navigate([`home/vendor/${this.idVendor}/store/${this.idStore}`]);
  };

  //=============Carga del mapa====================//

  public placeMarkerBranch(event: any) {
    this.latitude = event.coords.lat;
    this.longitude = event.coords.lng;
    this.selectedPoint = true;
    this.handleGetAddressGeo(this.latitude, this.longitude);
  }

  public async handleGetAddressGeo(lat: number, lng: number){
    let address = '';

    let geoCoder = new google.maps.Geocoder();
    let latLng = new google.maps.LatLng(lat, lng);
    const request: google.maps.GeocoderRequest = {
      location: latLng
    };

    await geoCoder.geocode(request, (results: string | any[])=>{
      address = results[0].formatted_address;
      this.entity.addressGeo = address;
    });
  }

  protected override initializeMap(): void {
    try {
      this.latitude = this.entity.latitude;
      this.longitude = this.entity.longitude;

    } catch (exception) {
      this.latitude = 0.0;
      this.longitude = 0.0;
    }
  }


  public async initializeCoordinates() {
    const paths: any[] = [];
    if (this.entity.coordinates) {
      await this.entity.coordinates?.forEach((locations: any) => {
        const points: any[] = [];
        locations.forEach((location:any) => {
          const latLng = new google.maps.LatLng(location[1], location[0]);
          points.push(latLng);
        });
        paths.push(points);
      });

      this.polygon = this.generatePolygon(paths, false);
      this.drawingManager.setDrawingMode(null);
      this.drawingManager.setOptions({drawingControl: false});
    }
  }


  public actionClean(): void {
    if (this.polygon) {
      this.polygon.setMap(null);
      this.initializeMap();
      //this.initializeDrawingManager();
      this.isClosedPolygon = false;
      this.isDrawInternalPolygon = false;
    }
  }

  protected override initializeDrawingManager = () => {
    if (this.drawingManager) {
      this.drawingManager.setDrawingMode(null);
      this.drawingManager.setOptions({drawingControl: false});
    }
    this.drawingManager = new google.maps.drawing.DrawingManager(this.drawingManagerOptions);

    this.drawingManager.setMap(this.map);
    google.maps.event.addListener(this.drawingManager, 'overlaycomplete',
      (event: any) => {
        this.polygonOverlayComplete(event);
      });
  };

}
