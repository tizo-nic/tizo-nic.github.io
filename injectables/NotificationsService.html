import { Injectable } from '@angular/core';

import { BehaviorSubject } from "rxjs";
import { SocketClient } from "../schemes/enum/socket-client";

// @ts-ignore
import io from 'socket.io-client';
import {TizoStorageService} from "./tizo-storage.service";
import {NgToastService} from "ng-angular-popup";

@Injectable({
  providedIn: 'root'
})
export class NotificationsService {

  public socket: any;
  public state: BehaviorSubject<SocketClient> | any;
  public amount: BehaviorSubject<any> = new BehaviorSubject<any>(0);
  public stateDelivery:BehaviorSubject<any> = new BehaviorSubject<any>({});

  constructor(
    private _tizoStorageService: TizoStorageService,
    private _toast: NgToastService
  ) {
    this.socket = io('https://notification.tizo.co/push-vendor', {
      transports: ['websocket'],
      query: {
        token: this._tizoStorageService.getTokenAuth()
      }
    });
    this.state = new BehaviorSubject<SocketClient>(SocketClient.ATTEMPTING);
    // this.socket.on('connect', (data: any) => {
    //   // console.log('connected', data);
    //   this.state.next(SocketClient.CONNECTED);
    // });

    // this.socket.on('connect_error', (err: any) => {
    //   this.state.next(SocketClient.ERROR);
    //   console.log(err, 'connect_error');
    // });
  }

  public notifyStatusDelivery(user: any){
    this.socket.on(user, ({content, data, typeMessage}: any) => {
      let {code, status, name} = data;
      if(typeMessage==='changeStatus'){
        if(status.name==='Rejected'){
          this._toast.warning({summary: '¡Aviso!', detail: content});
        }else{
          this._toast.success({summary: '¡Aviso!', detail: content});
        }
      }
      this.stateDelivery.next(data);
    });
  }

  public notifyQuantityDeliveries(user: any): any{
    this.socket.on(user, ({content, typeMessage} : any) => {
      if(typeMessage === 'countDeliveryRequest'){
        this.amount.next(content);
      }
    });
  }


}
