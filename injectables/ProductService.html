import { Injectable } from '@angular/core';
import {TizoRestService} from "../../../services/tizo-rest.service";
import {Observable} from "rxjs";
import {HttpResponse} from "@angular/common/http";

@Injectable({
  providedIn: 'root'
})
export class ProductService {

  constructor(
    private _tizoRestService: TizoRestService
  ) {
  }

  public getProducts(): Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/');
  }

  public getProductById(id: string):  Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/' + id);
  }

  public postProduct(product: any):  Observable<HttpResponse<any>> {
    return this._tizoRestService.post('/products/', product);
  }

  public putProduct(product: any, id: string):  Observable<HttpResponse<any>> {
    return this._tizoRestService.put(`/products/${id}/all`, product);
  }

  public deleteProduct(id: number):  Observable<HttpResponse<any>> {
    return this._tizoRestService.delete(`/products/${id}/`);
  }

  public getProductOptions():  Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/enums/options');
  }

  public getProductQualities():  Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/enums/qualities');
  }

  public getProductModelBrand():  Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/model-brand');
  }

  public postProductModelBrand(model: string, brand: string): Observable<HttpResponse<any>> {
    return this._tizoRestService.post('/products/model-brand', {model, brand});
  }

  public getAllCategories(): Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/categories/');
  }

  //---------------------------------------------------------------------------------//

  public getPromotionProduct(): Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/promotions/');
  }

  public postPromotionProduct(promotion: any): Observable<HttpResponse<any>> {
    return this._tizoRestService.post('/promotions/', promotion);
  }

  public getPromotionProductById(id: string): Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/promotions/' + id);
  }

  public deletePromotionProduct(id: string): Observable<HttpResponse<any>> {
    return this._tizoRestService.delete('/promotions/' + id);
  }

  //---------------------------------------------------------------------------------//
  public getProductOptionsList(): Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/enums/options');
  }

  public getStatsProduct(): Observable<HttpResponse<any>> {
    return this._tizoRestService.get('/products/stats');
  }

  //-----------------------------WholeSales Price--------------------------------//

  public getWholeSalesPrice(idProduct: string): Observable<HttpResponse<any>> {
    return this._tizoRestService.get(`/products/${idProduct}/wholesale-price/`);
  }

}
