import { Injectable } from '@angular/core';
import {environment} from '../../environments/environment';
import {HttpClient, HttpErrorResponse, HttpHeaders, HttpResponse} from '@angular/common/http';
import {catchError, map, Observable} from 'rxjs';
import {plainToClass} from 'class-transformer';
import {Router} from '@angular/router';
import {TizoStorageService} from "./tizo-storage.service";
import { ResponseRequest } from "../schemes/entities/response/response-request";

@Injectable({
  providedIn: 'root'
})
export class TizoRestService {
  private readonly SRV_HOST = environment.tizo.host;
  private readonly SRV_PATH = environment.tizo.apiPath;
  private readonly SRV_AUTH = environment.tizo.apiKey;

  private _httpHeaders: HttpHeaders;
  private _handleResponse: (httpResponse: HttpResponse<any>, metadata: any) => any;
  private _handleError: (httpErrorResponse: HttpErrorResponse) => any;

  constructor(
    protected readonly _router: Router,
    protected readonly _httpClient: HttpClient,
    private _tizoStorageService: TizoStorageService
  ) {
    this.handleResponse = this.getProcessResponse;
    this.handleError = this.getProcessError;
  }

  private getHost = (): string => this.SRV_HOST;


  public getApi = () => this.getHost() + '/' + this.SRV_PATH;



  private getProcessResponse(httpResponse: HttpResponse<any>, metadata: any = {}): any {
    const responseRequest: ResponseRequest = plainToClass(ResponseRequest, httpResponse);
    try{
      switch (responseRequest.status.code) {
      case 200:
        return responseRequest;
      case 400:
        console.log('400');
        return responseRequest;
      default:
        void this._router.navigate(['/signin']);
        return null;
      }
    } catch (e) {
      return null;
    }
  }

  protected getProcessError(httpErrorResponse: HttpErrorResponse): any {
    if (!navigator.onLine){
      return {
        status: {
          code: 0,
          message: 'No internet connection'
        }
      };
    }
    if (httpErrorResponse.error instanceof ErrorEvent) {
      console.log('error:', httpErrorResponse.error.message);
    } else {
      console.log(httpErrorResponse.error);
    }
    return httpErrorResponse;
  }

  /**
   * @description Set the headers for the request
   * @param str {string} - The header to set
   * @return {boolean} - True if the header was set
   * */
  public isJsonString(str: string): boolean {
    try {
      JSON.parse(str);
    } catch (e) {
      return false;
    }
    return true;
  }

  /**
   * @description Set the headers for the request
   * @returns {Observable<any>} - The response of the request
   */
  public get httpHeaders(): HttpHeaders {

    // Establecemos los headers, con el entorno a la hora de ejecutar la peticin
    let setHeadersEnvironment: string | { [name: string]: string | string[]} =
      {
        'Content-Type': 'application/json',
        'Authorization': this._tizoStorageService.getTokenAuth(),
        'environment': this._tizoStorageService.getDevEnvironmentNic(),
        'Accept-Language': 'es',
      };
    return this._httpHeaders = (!this._httpHeaders)
      ? new HttpHeaders(setHeadersEnvironment) : this._httpHeaders;
  }

  /**
   * @description set headers for the request
   * @param value - headers to set
   */
  public set httpHeaders(value: HttpHeaders) {this._httpHeaders = value;}

  /**
   * @description - Handle Http operation that failed.
   * @param httpResponse - name of the operation that failed
   * @param metadata - optional parameter
   */
  public processingResponse = (httpResponse: HttpResponse<any>, metadata:any  = {}): any => httpResponse;

  /**
   * @description - Handle response from server
   * @return {any} - returns the error
   * @param httpErrorResponse - the error from the server
   */
  public processingError = (httpErrorResponse: HttpErrorResponse): any => httpErrorResponse;

  /**
   * Obtiene una lista de errores
   * @return {Error}
   * */
  public get handleError(): (httpErrorResponse: HttpErrorResponse) => any {
    return this._handleError = (!this._handleError) ? this.processingError : this._handleError;
  }

  /**
   * Obtiene una lista de errores
   * @return {Error}
   * */
  public set handleError(value: (httpErrorResponse: HttpErrorResponse) => any) {this._handleError = value;}

  /**
   * Obtiene una lista de elementos
   * @return {Observable<any>}
   * */
  public get handleResponse (): (httpResponse: HttpResponse<any>, metadata: any) => any {
    return this._handleResponse = (this._handleResponse) ? this.processingResponse : this._handleResponse;
  }

  /**
   * Obtiene una lista de elementos
   * @return {Observable<any>}
   * */
  public set handleResponse(value: (httpResponse: HttpResponse<any>, metadata: any) => any) {
    this._handleResponse = value;
  }

  /**
   * Obtiene una lista de elementos con el servicio REST get
   * @param endpoint {string}
   * @param metadata {any}
   * @return {Observable<HttpResponse<any>>}
   * */
  public get(endpoint?: string, metadata: any = {}): Observable<HttpResponse<any>> | any {
    endpoint = this.getApi() + endpoint;
    return this._httpClient.get(endpoint, {headers: this.httpHeaders})
      .pipe(
        map((httpResponse: HttpResponse<any> | any) => this.handleResponse(httpResponse, metadata)),
        catchError(this.handleError)
      );
  }

  public post(endpoint: string, body: any, metadata: any = {}): Observable<HttpResponse<any>> | any {
    endpoint = this.getApi() + endpoint;

    let setHeadersEnvironment: any =
      {
        'Content-Type': 'application/json',
        'Authorization': this._tizoStorageService.getTokenAuth(),
        'environment': this._tizoStorageService.getDevEnvironmentNic(),
        'Accept-Language': 'es',
      };

    this.httpHeaders = setHeadersEnvironment;

    return this._httpClient.post(endpoint, body, {headers: this.httpHeaders})
      .pipe(
        map((httpResponse: HttpResponse<any> | any) => this.handleResponse(httpResponse, metadata)),
        catchError(this.handleError)
      );
  }

  public put(endpoint: string, body: any, metadata: any = {}): Observable<HttpResponse<any>> | any {
    endpoint = this.getApi() + endpoint;
    return this._httpClient.put(endpoint, body, {headers: this.httpHeaders})
      .pipe(
        map((httpResponse: HttpResponse<any> | any) => this.handleResponse(httpResponse, metadata)),
        catchError(this.handleError)
      );
  }

  public delete(endpoint: string, metadata: any = {}): Observable<HttpResponse<any>> | any {
    endpoint = this.getApi() + endpoint;
    return this._httpClient.delete(endpoint, {headers: this.httpHeaders})
      .pipe(
        map((httpResponse: HttpResponse<any> | any) => this.handleResponse(httpResponse, metadata)),
        catchError(this.handleError)
      );
  }

  public postAuth(endpoint: string, body: any, metadata: any = {}): Observable<HttpResponse<any>> | any {
    endpoint = this.getApi() + endpoint;
    let setHeadersEnvironment: any =
      {
        'Content-Type': 'application/json',
        'Authorization': this._tizoStorageService.getTokenAuth(),
        'environment': this._tizoStorageService.getDevEnvironmentNic(),
        'Accept-Language': 'es',
      };

    this.httpHeaders = setHeadersEnvironment;

    return this._httpClient.post(endpoint, body,{headers: this.httpHeaders} )
      .pipe(
        map((httpResponse: HttpResponse<any> | any) => this.handleResponse(httpResponse, metadata)),
        catchError(this.handleError)
      );
  }

  /**
   * @description - Upload a file to the server
   * @param endpoint - The endpoint of the request
   * @param body - The file to upload type File
   * @param metadata - optional parameter
   * */
  public async postUploadFiles(endpoint: string, body: any, metadata: any = {}) {
    let {name} = body;
    let result_: string = '';

    let myHeaders = new Headers();
    myHeaders.append("Authorization", this._tizoStorageService.getTokenAuth());
    myHeaders.append("environment", this._tizoStorageService.getEnviorment());

    let formData = new FormData();
    formData.append("file", body, name);

    let requestOptions:any = {
      method: "POST",
      headers: myHeaders,
      body: formData,
      redirect: "follow",
    };

    await fetch(endpoint, requestOptions)
      .then((response) => response.text())
      .then((result) => {
        result_ = result;
      })
      .catch((error) =>{
        console.log("error", error);
        result_ = '';
      });

    return result_;
  }



}
