import { Injectable } from '@angular/core';
import { TizoRestService } from "../tizo-rest.service";
import { Observable } from "rxjs";
import { HttpResponse } from "@angular/common/http";

@Injectable({
  providedIn: 'root'
})
export class RateListRestService {

  constructor(private _tizoRestService: TizoRestService,) {}

  public getRateList = ():Observable<HttpResponse<any>> => this._tizoRestService.get(`/system-data/rate-list/`);

  public getRateListById = (id: string): Observable<HttpResponse<any>> => this._tizoRestService.get(`/system-data/rate-list/${id}/`);

  public postRateListIdCapacity = (idCapacity: string, bodyRateList: any): Observable<HttpResponse<any>> => this._tizoRestService.post(`/system-data/capacities/${idCapacity}/rate-list/`, bodyRateList);

  public postRateListItemIdRateList = (idRateList: string, bodyRateListItem: any): Observable<HttpResponse<any>> => this._tizoRestService.post(`/system-data/rate-list/${idRateList}/items/`, bodyRateListItem);

  public putRateList = (id: string, data: any) => this._tizoRestService.put(`/system-data/rate-list/${id}/`, data);

  public deleteRateList = (id: string) => this._tizoRestService.delete(`/system-data/rate-list/${id}/`);

  public deleteRateListItem = (idRateList: string, idRateItem: string) => this._tizoRestService.delete(`/system-data/rate-list/${idRateList}/item/${idRateItem}/`);

  public getRateListItem = (id: string):Observable<HttpResponse<any>> => this._tizoRestService.get(`/system-data/rate-list/${id}/items/`);

  public getRateListItemById = (idElementRateList: string, idElementItem: string): Observable<HttpResponse<any>> => this._tizoRestService.get(`/system-data/rate-list/${idElementRateList}/item/${idElementItem}`);

  public putRateListItemById = (idElementRateList: string, idElementItem: string, bodyRateListItem: any): Observable<HttpResponse<any>> => this._tizoRestService.put(`/system-data/rate-list/${idElementRateList}/item/${idElementItem}`, bodyRateListItem);
}
