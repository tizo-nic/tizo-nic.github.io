import { Injectable } from '@angular/core';

declare const google: any;

@Injectable({
  providedIn: 'root'
})
export class MapService {

  /**
   * Realiza una copia del mapa
   * @returns mapa copiado
   * @param object
   * */
  public copy(object: any): any{
    return JSON.parse(JSON.stringify(object));
  }

  /**
  * Realiza una extensi贸n del mapa
  * @param object1
  * @param object2
  * @returns object1 extendido con object2
  * */
  public extend = (object1: any, object2: any): any => Object.assign(object1, object2);


  /**
  * Quita los marcadores del mapa
  * @param object:
  * @returns object sin marcadores
  * */
  public removeNulls(object: any): any {
    object = this.copy(object);
    Object.keys(object).forEach((key) => (object[key] === null) && delete object[key]);
    return object;
  }

  /**
   * Calcula la distancia entre dos puntos
   * @param origenLat: latitud del punto 1
   * @param origenLong: longitud del punto 1
   * @param destinoLat: latitud del punto 2
   * @param destinoLong: longitud del punto 2
   * @returns distancia entre los puntos
   * */
  public distance(origenLat: any, origenLong: any, destinoLat: any, destinoLong: any) {
    let origin = new google.maps.LatLng(origenLat, origenLong);
    let destination = new google.maps.LatLng(destinoLat, destinoLong);
    const distance = google.maps.geometry.spherical.computeDistanceBetween(origin, destination) / 1000;
    return distance.toFixed(2);
  }


  /**
   * Obtiene la posici贸n actual del punto en el mapa
   * @param Promise: any
   * */
  public getPosition(): Promise<any> {
    return new Promise((resolve, reject) => {
      navigator.geolocation.getCurrentPosition(
        resp => {resolve({lng: resp.coords.longitude, lat: resp.coords.latitude});
        }, err => {reject(err);});
    });

  }




  constructor() { }
}
