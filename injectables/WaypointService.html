import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable } from "rxjs";
import { WaypointModel } from "../../models/delivery/waypoint.model";

@Injectable({
  providedIn: 'root'
})
export class WaypointService {

  public wayPoints: WaypointModel[] = [];
  public wayPoints$: BehaviorSubject<WaypointModel[]> = new BehaviorSubject<WaypointModel[]>(this.wayPoints);
  public isNewWayPoint: boolean;

  constructor() {}

  public addWayPoint(wayPoint: WaypointModel) {
    this.wayPoints.push(wayPoint);
    this.wayPoints$.next(this.wayPoints);
  }

  public getWayPoints = (): Observable<any> =>
    this.wayPoints$.asObservable();

  public editWayPoint(id:number, wayPoint: WaypointModel) {
    // edit item of list by id
    this.wayPoints[id] = wayPoint;
    this.wayPoints$.next(this.wayPoints);
  }

  public changeOrderWayPoints(newWayPoints: WaypointModel[]) {
    this.wayPoints = newWayPoints;
    this.wayPoints$.next(this.wayPoints);
  }

  public deleteWayPoint(id:number){
    // delete item of list by id
    this.wayPoints.splice(id, 1);
    this.wayPoints$.next(this.wayPoints);
  }

  //get point by id
  public getPointById = (id: number) =>{
    return this.wayPoints[id];
  };

  // Clean list of points
  public cleanListPoints = () =>{
    this.wayPoints = [];
    this.wayPoints$.next(this.wayPoints);
  };


  public setWayPointB(wayPointB: WaypointModel){
    this.wayPoints[1] = wayPointB;
    this.wayPoints$.next(this.wayPoints);
  }

  public setWayPointA(wayPointA: WaypointModel){
    this.wayPoints[0] = wayPointA;
    this.wayPoints$.next(this.wayPoints);
  }

  public switchNewWayPoint(state: boolean){
    this.isNewWayPoint = state;
  }

}
