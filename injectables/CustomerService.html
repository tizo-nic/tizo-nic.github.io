import { Injectable } from '@angular/core';
import {HttpResponse} from "@angular/common/http";
import {TizoRestService} from "../../services/tizo-rest.service";
import {CustomerModel} from './customer.model';
import {BehaviorSubject, Observable} from "rxjs";

@Injectable({
  providedIn: 'root'
})
export class CustomerService {

  //guardar clientes
  public saveCustomers: boolean[] = [false, false];
  public customersSave$: BehaviorSubject<boolean[]> = new BehaviorSubject<boolean[]>(this.saveCustomers);

  public customerPointSubject: BehaviorSubject<CustomerModel> = new BehaviorSubject<CustomerModel>({} as CustomerModel);
  public readonly customerPoint$: Observable<CustomerModel> = this.customerPointSubject.asObservable();  

  constructor(
    private _tizoRestService :TizoRestService
  ) { }

  /*------------------ Procesos de control para guardar clientes ---------------------*/

  public setCustomersSave(id: number, isSave: boolean) {
    this.saveCustomers[id] = isSave;
    this.customersSave$.next(this.saveCustomers);
  }

  public resetCustomersSave(): void {
    this.saveCustomers = [false, false];
    this.customersSave$.next(this.saveCustomers);
  }

  /*------------------ Obtener punto del cliente ---------------------*/

  public getCustomerPoint(): Observable<any> {
    return this.customerPointSubject;
  }

  public setCustomerPoint(customerPoint: CustomerModel): void {
    this.customerPointSubject.next(customerPoint);
  }


  public getAllCustomers = (idStore: string):Observable<HttpResponse<any>> => this._tizoRestService.get(`/vendors/stores/${idStore}/customers/`);

  public getCustomerById = (idStore: string, idCustomer: string):Observable<HttpResponse<any>> => this._tizoRestService.get(`/vendors/stores/${idStore}/customers/${idCustomer}/`);

  public putCustomer = (idStore: string, idCustomer: string, body: any):Observable<HttpResponse<any>> => this._tizoRestService.put(`/vendors/stores/${idStore}/customers/${idCustomer}/`, body);

  public postCustomer = (idStore: string, body: any):Observable<HttpResponse<any>> => this._tizoRestService.post(`/vendors/stores/${idStore}/customers/`, body);

  public deleteCustomer = (idStore: string, idCustomerAddress: string):Observable<HttpResponse<any>> => this._tizoRestService.delete(`/vendors/stores/${idStore}/customers/${idCustomerAddress}/`);

}
