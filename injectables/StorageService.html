import {Injectable} from '@angular/core';

@Injectable()
export class StorageService {

  protected _encodeData: (value: string) => string;
  protected _decodeData: (value: string) => string;

  constructor() {
    this._encodeData = this.encodeData;
    this._decodeData = this.decodeData;
  }

  protected getStorage(key: string): any {
    const storage: string | any = localStorage.getItem(key);
    if (storage) {
      return JSON.parse(this._decodeData(storage));
    } else {
      return {data: {}};
    }
  }

  protected setStorage(key: string, storage: any): void {
    localStorage.setItem(key, this._encodeData(JSON.stringify(storage)));
  }

  protected removeStorage(key: string): void {
    localStorage.removeItem(key);
  }

  protected addAttribute(data: any, key: string): void {
    const storage: any = this.getStorage(key);
    if (storage && storage.data) {
      Object.assign(storage.data, data);
      this.setStorage(key, storage);
    }
  }

  protected getAttribute(attribute: string, key: string): any {
    const storage: any = this.getStorage(key);
    if (this.attributeExists(attribute, key)) {
      return storage.data[attribute];
    } else {
      return null;
    }
  }

  protected removeAttribute(attribute: string, key: string): void {
    const storage: any = this.getStorage(key);
    if (this.attributeExists(attribute, key)) {
      delete storage.data[attribute];
      this.setStorage(key, storage);
    }
  }

  public attributeExists(attribute: string, key: string): boolean {
    return !!this.getStorage(key).data[attribute];
  }

  protected encodeData(value: string): string {
    return value;
  }

  protected decodeData(value: string): string {
    return value;
  }
}
