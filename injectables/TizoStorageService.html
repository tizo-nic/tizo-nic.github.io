import { Injectable } from '@angular/core';
import {BehaviorSubject} from "rxjs";

@Injectable({
  providedIn: 'root'
})
export class TizoStorageService {

  public needToShow$: BehaviorSubject<boolean> = new BehaviorSubject<boolean>(true);

  constructor() {

  }

  //logica para identificar si se necesita mostrar el modal de los terminos y condiciones
  public setFirstShow(value: boolean) {
    this.needToShow$.next(value);
  }

  public getTokenAuth = (): string => localStorage.getItem('auth_token') ?? '';
  public getEnviorment = (): string => localStorage.getItem('environment') ?? '';

  public setDevEnvironmentNic = (country: number): void => {
    let countryNic = (country === 0) ? 'dev' : (country === 1) ? 'ni' : 'pa';
    localStorage.setItem('environment', countryNic);
  };

  public setIndexLastBranchSelected = (index: number): void => {
    localStorage.setItem('indexLastBranchSelected', index.toString());
  };

  public isAcceptedTerms = (): string => localStorage.getItem('AcceptedTerms') ?? '';
  public setAcceptedTerms = (value: string): void => localStorage.setItem('AcceptedTerms', value);

  public getindexLastBranchSelected = (): number => parseInt(localStorage.getItem('indexLastBranchSelected') ?? '0');
  public getDevEnvironmentNic = (): string => localStorage.getItem('environment') || '';
}
