import { Injectable } from '@angular/core';
import {TizoRestService} from "../tizo-rest.service";
import {HttpResponse} from "@angular/common/http";
import {Observable} from "rxjs";

@Injectable({
  providedIn: 'root'
})
export class AreaRestService {

  constructor(
    private tizoRestService: TizoRestService
  ) { }

  public getAllAreas = (): Observable<HttpResponse<any>> => this.tizoRestService.get('/system-data/areas/');

  public getAreaById = (id: string): Observable<HttpResponse<any>> => this.tizoRestService.get(`/system-data/areas/${id}/`);

  public postArea = (area: any): Observable<HttpResponse<any>> => this.tizoRestService.post('/system-data/areas/', area);

  public putArea = (idArea: any, area: any): Observable<HttpResponse<any>> =>
    this.tizoRestService.put(`/system-data/areas/${idArea}/`, area);

  public deleteArea = (id: number): Observable<HttpResponse<any>> => this.tizoRestService.delete(`/system-data/areas/${id}/`);

  /*==================================================================*/

  public getAllDepartment = (): Observable<HttpResponse<any>> => this.tizoRestService.get(`/system-data/departaments/`);

  public getDepartmentsByIdCountry = (): Observable<HttpResponse<any>> => this.tizoRestService.get(`/system-data/departments/`);

  public getDepartmentByIdCountry = (idCountry: number, idDepartment: number): Observable<HttpResponse<any>> =>
    this.tizoRestService.get(`/system-data/countries/${idCountry}/departments/${idDepartment}/`);

  public postDepartment = (idCountry:number, department: any): Observable<HttpResponse<any>> => this.tizoRestService.post(`/system-data/countries/${idCountry}/departments/`, department);

  /*==================================================================*/

  public getMunicipalityByIdDepartment = (idDepartment: number): Observable<HttpResponse<any>> => this.tizoRestService.get(`/system-data/departments/${idDepartment}/municipalities/`);



}
