import { Injectable } from '@angular/core';
import {TizoRestService} from "../../../services/tizo-rest.service";
import {HttpResponse} from "@angular/common/http";
import {Observable} from "rxjs";

@Injectable({
  providedIn: 'root'
})
export class StoreRestService {

  constructor(
    public _tizoRestService: TizoRestService
  ) {
  }

  public getAllStores = (): Observable<HttpResponse<any>> =>
    this._tizoRestService.get('/vendors/stores/');

  public getStoreById = (idStore: string): Observable<HttpResponse<any>> =>
    this._tizoRestService.get(`/vendors/stores/${idStore}/`);

  public postStore = (data: any): Observable<HttpResponse<any>> =>
    this._tizoRestService.post('/vendors/stores/', data);

  public putStore = (idStore: string, data: any): Observable<HttpResponse<any>> =>
    this._tizoRestService.put(`/vendors/stores/${idStore}/`, data);

  public deleteStore = (idStore: string): Observable<HttpResponse<any>> =>
    this._tizoRestService.delete(`/vendors/stores/${idStore}/`);

  public uploadImage = (idStore: any,storeName: string, file: any) =>{
    return this._tizoRestService.postUploadFiles(`https://api.tizo.co/api/v1/files/uploadFile/?folderName=documents-vendor-${idStore}`, file);
  };

}
