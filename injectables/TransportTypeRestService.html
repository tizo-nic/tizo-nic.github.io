import {TizoRestService} from "../tizo-rest.service";
import {Injectable} from "@angular/core";
import {Observable} from "rxjs";
import {HttpResponse} from "@angular/common/http";

@Injectable({
  providedIn: 'root'
})
export class TransportTypeRestService {
  constructor(
    private _tizoRestService: TizoRestService,
  ) { }

  public getTransportTypes = (): Observable<HttpResponse<any>> => this._tizoRestService.get(`/system-data/transport-types/`);

  public getTransportTypeById = (id: string): Observable<HttpResponse<any>> => this._tizoRestService.get(`/system-data/transport-types/${id}`);

  public postTransportType = (transportType: any): Observable<HttpResponse<any>> => this._tizoRestService.post(`/system-data/transport-types/`, transportType);

  public putTransportType = (id: string, transportType: any): Observable<HttpResponse<any>> => this._tizoRestService.put(`/system-data/transport-types/${id}`, transportType);

  public deleteTransportType = (id: string): Observable<HttpResponse<any>> => this._tizoRestService.delete(`/system-data/transport-types/${id}`);

}
