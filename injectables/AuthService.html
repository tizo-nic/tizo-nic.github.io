import { Injectable } from '@angular/core';

import { TizoRestService } from '../services/tizo-rest.service';
import {Observable} from "rxjs";
import {HttpResponse} from "@angular/common/http";

@Injectable({
  providedIn: 'root'
})
export class AuthService {

  constructor(
    private tizoRestService: TizoRestService
  ) {}

  /**
   * Set token for the user
   */
  public setToken = (token: string) => localStorage.setItem('auth_token', token);

  /**
   * Close session of the user vendor
   * @returns {string}
   */
  public authSignOut = () =>localStorage.removeItem('auth_token');


  /**
   * @description - Delete token from localStorage
   * */
  public authSignOutEnvironment = () =>localStorage.removeItem('environment');

  /**
   * Get token of the user
   * @returns {string}
   */
  public authSignIn = (bodySignIn: any): Observable<HttpResponse<any>> =>
    this.tizoRestService.postAuth('/auth/login/', bodySignIn);

}
