import { FormSegment } from '../../support/form-segment';

// Interface
import { FormInterface } from '../../interface/form.interface';
import { TableInterface } from '../../interface/table.interface';

// Templates for the component
import { FormTemplate } from '../../support/form-template';
import { TableTemplate } from '../../support/table-template';

export class AreaEntity implements FormInterface, TableInterface {

  public idArea: number;
  public idDepartment: number;
  public idMunicipality: number;
  public name: string;
  public description: string;
  public geojson: string;
  public isActive: boolean;
  public dateCreation: Date;
  public dateLastUpdate: Date;
  public coordinates: [[number[]]];

  public Departament: {
    idDepartament: number,
    idCountry: number,
    name: string,
    latitude: number,
    longitude: number,
    isActive: boolean,
    dateCreation: Date,
    dateLastUpdate: Date,
    active: boolean,
  };

  public Municipalities: any;

  public file: string;
  public tableTemplates: TableTemplate[];
  public formSegments: FormSegment[];

  public setTableTemplate(type: string): TableTemplate[] {
    switch (type) {
    case 'simple': return this.getTableTemplateSimple();
    case 'simple-department': return this.getTableTemplateSimpleDepartment();
    default:
      return [];
    }
  }

  public getTableTemplateSimpleDepartment(): TableTemplate[] {
    const tableTemplates: TableTemplate[] = [];

    const tableTemplateOptions: TableTemplate = new TableTemplate();
    tableTemplateOptions.alignment = 'center';
    tableTemplateOptions.cssClass = '';
    tableTemplateOptions.caption = 'Opciones';
    tableTemplateOptions.cellTemplate = 'options';
    tableTemplateOptions.priority = 9;
    tableTemplateOptions.allowExporting = false;
    tableTemplates.push(tableTemplateOptions);

    const tableTemplateName: TableTemplate = new TableTemplate();
    tableTemplateName.cssClass = 'dx-column-middle';
    tableTemplateName.alignment = 'center';
    tableTemplateName.caption = 'Nombre';
    tableTemplateName.dataField = 'name';
    tableTemplateName.dataType = 'string';
    // tableTemplateName.sortOrder = 'asc';
    tableTemplateName.priority = 8;
    tableTemplateName.allowExporting = true;
    tableTemplates.push(tableTemplateName);


    return tableTemplates;
  }

  public getTableTemplateSimple(): TableTemplate[] {

    const tableTemplates: TableTemplate[] = [];

    const tableTemplateOptions: TableTemplate = new TableTemplate();
    tableTemplateOptions.width = 140;
    tableTemplateOptions.alignment = "center";
    tableTemplateOptions.cssClass = "dx-column-middle";
    tableTemplateOptions.caption = "Opciones";
    tableTemplateOptions.cellTemplate = "options";
    tableTemplateOptions.priority = 6;
    tableTemplateOptions.allowExporting = false;
    tableTemplates.push(tableTemplateOptions);



    const tableTemplateName: TableTemplate = new TableTemplate();
    tableTemplateName.cssClass = "dx-column-middle";
    tableTemplateName.caption = "Nombre";
    tableTemplateName.dataField = "name";
    tableTemplateName.dataType = "string";
    tableTemplateName.cellTemplate = "bold";
    tableTemplateName.priority = 4;
    tableTemplateName.allowExporting = true;
    tableTemplates.push(tableTemplateName);

    const tableTemplateDescription: TableTemplate = new TableTemplate();
    tableTemplateDescription.cssClass = "dx-column-middle";
    tableTemplateDescription.caption = "Descripción";
    tableTemplateDescription.dataField = "description";
    tableTemplateDescription.dataType = "string";
    tableTemplateDescription.priority = 3;
    tableTemplateDescription.allowExporting = true;
    tableTemplates.push(tableTemplateDescription);

    const tableTemplateDateCreation: TableTemplate = new TableTemplate();
    tableTemplateDateCreation.width = 200;
    tableTemplateDateCreation.cssClass = "dx-column-middle";
    tableTemplateDateCreation.caption = "Fecha de creación";
    tableTemplateDateCreation.dataField = "dateCreation";
    tableTemplateDateCreation.dataType = "date";
    tableTemplateDateCreation.priority = 2;
    tableTemplateDateCreation.allowExporting = true;
    tableTemplates.push(tableTemplateDateCreation);

    const tableTemplateDateUpdate: TableTemplate = new TableTemplate();
    tableTemplateDateUpdate.width = 200;
    tableTemplateDateUpdate.cssClass = "dx-column-middle";
    tableTemplateDateUpdate.caption = "Fecha de actualización";
    tableTemplateDateUpdate.dataField = "dateLastUpdate";
    tableTemplateDateUpdate.dataType = "date";
    tableTemplateDateUpdate.priority = 1;
    tableTemplateDateUpdate.allowExporting = true;
    tableTemplates.push(tableTemplateDateUpdate);

    const tableTemplateIdCountry: TableTemplate = new TableTemplate();
    tableTemplateIdCountry.width = 200;
    tableTemplateIdCountry.cssClass = "dx-column-middle";
    tableTemplateIdCountry.caption = "text_Id_country";
    tableTemplateIdCountry.dataField = "idCountry";
    tableTemplateIdCountry.dataType = "string";
    tableTemplateIdCountry.priority = 7;
    tableTemplateIdCountry.allowExporting = true;
    tableTemplateIdCountry.hidden = false;
    tableTemplates.push(tableTemplateIdCountry);

    const tableTemplatePoints: TableTemplate = new TableTemplate();
    tableTemplatePoints.width = 200;
    tableTemplatePoints.cssClass = "dx-column-middle";
    tableTemplatePoints.caption = "text_coordinates";
    tableTemplatePoints.dataField = "points";
    tableTemplatePoints.dataType = "string";
    tableTemplatePoints.priority = 7;
    tableTemplatePoints.allowExporting = true;
    tableTemplatePoints.hidden = false;
    tableTemplates.push(tableTemplatePoints);

    return tableTemplates;
  }

  public setFormSegment(type: string): FormSegment[] {
    switch (type) {
    case 'simple': return this.getFormSegmentSimple();
    case "modalform": return this.getFormSegmentModal();
    default: return [];
    }
  }

  public getFormSegmentSimple(): FormSegment[] {

    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = "Información general de áreas.";
    formSegmentGeneralInformation.layoutCss = 'grid grid-cols-2 pt-3';
    formSegmentGeneralInformation.templates = [];

    const formTemplateName: FormTemplate = new FormTemplate();
    formTemplateName.field = "name";
    formTemplateName.name = "Nombre";
    formTemplateName.placeholder = "ej: principal";
    formTemplateName.type = "simple-text";
    formTemplateName.isRequired = true;
    formTemplateName.cssClass = "col-md-6";
    formTemplateName.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateName);

    const formTemplateDescription: FormTemplate = new FormTemplate();
    formTemplateDescription.field = "description";
    formTemplateDescription.name = "Descripción";
    formTemplateDescription.placeholder = "Principal";
    formTemplateDescription.type = "simple-text";
    formTemplateDescription.isRequired = true;
    formTemplateDescription.cssClass = "col-md-12";
    formTemplateDescription.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateDescription);




    formSegments.push(formSegmentGeneralInformation);

    return formSegments;
  }

  public getFormSegmentModal(): FormSegment[] {
    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = "text_general_information";
    formSegmentGeneralInformation.templates = [];

    const formTemplateCountry: FormTemplate = new FormTemplate();
    formTemplateCountry.field = "idCountry";
    formTemplateCountry.name = "text_country";
    formTemplateCountry.placeholder = "text_country";
    formTemplateCountry.type = "simple-selector";
    formTemplateCountry.isRequired = true;
    formTemplateCountry.cssClass = "col-md-12";
    formSegmentGeneralInformation.templates.push(formTemplateCountry);

    const formTemplatefile: FormTemplate = new FormTemplate();
    formTemplatefile.field = "file";
    formTemplatefile.name = "text_file";
    formTemplatefile.placeholder = "text_upload_document";
    formTemplatefile.type = "simple-file";
    formTemplatefile.isRequired = true;
    formTemplatefile.cssClass = "col-md-12";
    formTemplatefile.messageSmall = "allowed_documents";
    formSegmentGeneralInformation.templates.push(formTemplatefile);
    formSegments.push(formSegmentGeneralInformation);
    return formSegments;
  }



}
