import { FormInterface } from "../../interface/form.interface";
import { TableInterface } from "../../interface/table.interface";
import { FormSegment } from "../../support/form-segment";
import { TableTemplate } from "../../support/table-template";
import { ProfileManagerModel } from "../../models/profileManager.model";
import { FormTemplate } from "../../support/form-template";

export class ProfileManagerEntity implements ProfileManagerModel, FormInterface, TableInterface{
  idProfileVendorUser: number;
  description: string;
  name: string;
  active: boolean;
  dateCreation: Date;
  dateLastUpdate: Date;


  formSegments: FormSegment[];
  tableTemplates: TableTemplate[];

  public setFormSegment(type: string, fieldShow?: boolean): FormSegment[] {
    switch (type) {
    case 'simple':
      return this.getFormSegmentViewProfileVendor();
    default:
      return [];
    }
  }

  public getFormSegmentViewProfileVendor(): FormSegment[] {

    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = 'Perfil del vendor';
    formSegmentGeneralInformation.layoutCss = 'grid grid-cols-2 pt3';
    formSegmentGeneralInformation.templates = [];

    const formTemplateName: FormTemplate = new FormTemplate();
    formTemplateName.field = 'name';
    formTemplateName.name = 'Nombre';
    formTemplateName.placeholder = 'Nombre del perfil';
    formTemplateName.type = 'simple-text';
    formTemplateName.isRequired = true;
    formTemplateName.cssClass = 'my-2';
    formTemplateName.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateName);

    const formTemplateDescription: FormTemplate = new FormTemplate();
    formTemplateDescription.field = 'description';
    formTemplateDescription.name = 'Descripción';
    formTemplateDescription.placeholder = 'Descripción del perfil';
    formTemplateDescription.type = 'simple-text';
    formTemplateDescription.isRequired = true;
    formTemplateDescription.cssClass = 'my-2';
    formTemplateDescription.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateDescription);

    formSegments.push(formSegmentGeneralInformation);

    return formSegments;
  }

  public setTableTemplate(type: string): TableTemplate[] {
    return [];
  }

}
