import {IStoreModel} from "../models/store.model";
import {FormInterface} from "../../../schemes/interface/form.interface";
import {TableInterface} from "../../../schemes/interface/table.interface";
import {FormSegment} from "../../../schemes/support/form-segment";
import {TableTemplate} from "../../../schemes/support/table-template";
import {FormTemplate} from "../../../schemes/support/form-template";

export class StoreEntity implements IStoreModel,  FormInterface, TableInterface {
  active: boolean;
  creditBag: any;
  dateCreation: Date;
  dateLastUpdate: Date;
  dba: string;
  description: string;
  email: string;
  idStore: number;
  isPrivate: boolean;
  name: string;
  notifications: any;
  pathImage: string;
  phone: string;
  ruc: string;
  vacation: boolean;
  formSegments: FormSegment[];
  tableTemplates: TableTemplate[];

  public setFormSegment(type: string, fieldShow?: boolean): FormSegment[] {
    switch (type) {
    case 'simple':
      return this.getSimpleFormSegments();
    default:
      return [];
    }
  }

  public getSimpleFormSegments(): FormSegment[] {
    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = 'Información de la tienda';
    formSegmentGeneralInformation.layoutCss = 'grid grid-cols-2 pt-3';
    formSegmentGeneralInformation.templates = [];

    const formTemplateName: FormTemplate = new FormTemplate();
    formTemplateName.field = 'name';
    formTemplateName.name = 'Nombre';
    formTemplateName.placeholder = 'Nombre de la empresa';
    formTemplateName.type = 'simple-text';
    formTemplateName.isRequired = true;
    formTemplateName.cssClass = 'my-2';
    formTemplateName.icon = 'ri-user-line';
    formSegmentGeneralInformation.templates.push(formTemplateName);

    const formTemplateDBA: FormTemplate = new FormTemplate();
    formTemplateDBA.field = 'dba';
    formTemplateDBA.name = 'Razón social';
    formTemplateDBA.placeholder = 'Razón social';
    formTemplateDBA.type = 'simple-text';
    formTemplateDBA.isRequired = true;
    formTemplateDBA.cssClass = 'my-2';
    formTemplateDBA.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateDBA);

    const formTemplateEmail: FormTemplate = new FormTemplate();
    formTemplateEmail.field = 'email';
    formTemplateEmail.name = 'Email';
    formTemplateEmail.placeholder = 'Email de la empresa';
    formTemplateEmail.type = 'simple-email';
    formTemplateEmail.isRequired = true;
    formTemplateEmail.cssClass = 'my-2';
    formTemplateEmail.icon = 'ri-mail-line';
    formSegmentGeneralInformation.templates.push(formTemplateEmail);

    const formTemplatePhone: FormTemplate = new FormTemplate();
    formTemplatePhone.field = 'phone';
    formTemplatePhone.name = 'Teléfono';
    formTemplatePhone.placeholder = 'Teléfono de la empresa';
    formTemplatePhone.type = 'simple-phone';
    formTemplatePhone.isRequired = true;
    formTemplatePhone.cssClass = 'my-2';
    formTemplatePhone.icon = 'ri-smartphone-line';
    formSegmentGeneralInformation.templates.push(formTemplatePhone);

    const formTemplateDescription = new FormTemplate();
    formTemplateDescription.field = 'description';
    formTemplateDescription.name = 'Descripción';
    formTemplateDescription.placeholder = 'Descripción de la empresa';
    formTemplateDescription.type = 'simple-text';
    formTemplateDescription.isRequired = false;
    formTemplateDescription.cssClass = 'my-2';
    formTemplateDescription.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateDescription);


    const formTemplateRUC = new FormTemplate();
    formTemplateRUC.field = 'ruc';
    formTemplateRUC.name = 'RUC';
    formTemplateRUC.placeholder = 'RUC de la empresa';
    formTemplateRUC.type = 'simple-text';
    formTemplateRUC.readonly = true;
    formTemplateRUC.isRequired = true;
    formTemplateRUC.cssClass = 'my-2';

    formTemplateRUC.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateRUC);


    // const formTemplateTraditional = new FormTemplate();
    // formTemplateTraditional.field = 'vacation';
    // formTemplateTraditional.name = 'Vacación';
    // formTemplateTraditional.placeholder = 'Tradicional del transportista';
    // formTemplateTraditional.type = 'simple-radio';
    // formTemplateTraditional.readonly = false;
    // formTemplateTraditional.isRequired = true;
    // formTemplateTraditional.cssClass = 'my-2';
    // formSegmentGeneralInformation.templates.push(formTemplateTraditional);

    formSegments.push(formSegmentGeneralInformation);

    return formSegments;
  }

  public setTableTemplate(type: string): TableTemplate[] {
    return [];
  }

}
