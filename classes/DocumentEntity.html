  public setTableTemplate(type: string): TableTemplate[] {
    switch (type) {
    case 'simple':
      return this.getTableTemplateSimple();
    case 'simpleTizoVoyAndCargo':
      return this.getTableTemplateSimpleTizoVoyAndCargo();
    default:
      return [];

  }
