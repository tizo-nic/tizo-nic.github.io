import {FormInterface} from '../../interface/form.interface';
import {TableInterface} from '../../interface/table.interface';

//Templates
import {TableTemplate} from '../../support/table-template';
import {FormSegment} from '../../support/form-segment';
import {FormTemplate} from '../../support/form-template';
import { CapacityModel } from "../../models/Capacity.model";
import {MeasurementEntity} from './measurement.entity';

export class CapacityEntity implements CapacityModel,FormInterface, TableInterface{

  idCapacity: number;
  name: string;
  code: string;
  maxWeight: number;
  maxDepth: number;
  maxWidth: number;
  maxHeight: number;

  measurement: MeasurementEntity;

  public active: boolean;
  public dateCreation: Date;
  public dateLastUpdate: Date;

  public formSegments: FormSegment[];
  public tableTemplates: TableTemplate[];


  /**
   * Establece el formato visual de la tabla
   * @param typeTemplate tipo de tabla
   * @return {TableTemplate}
   */

  public setTableTemplate(typeTemplate: string): TableTemplate[] {
    switch (typeTemplate) {
    case 'simple':
      return this.getTableTemplateViewCapacity();
    default:
      return [];
    }
  }

  /**
   * Obtiene el formato visual de la tabla
   * @return {TableTemplate}
   */
  public getTableTemplateViewCapacity(): TableTemplate[] {
    const tableTemplates: TableTemplate[] = [];

    const tableTemplateOptions: TableTemplate = new TableTemplate();
    tableTemplateOptions.alignment = 'center';
    tableTemplateOptions.cssClass = '';
    tableTemplateOptions.caption = 'Opciones';
    tableTemplateOptions.cellTemplate = 'options';
    tableTemplateOptions.priority = 10;
    tableTemplateOptions.allowExporting = false;
    tableTemplates.push(tableTemplateOptions);

    const tableTemplateName: TableTemplate = new TableTemplate();
    tableTemplateName.cssClass = 'dx-column-middle';
    tableTemplateName.alignment = 'center';
    tableTemplateName.caption = 'Nombre';
    tableTemplateName.dataField = 'name';
    tableTemplateName.dataType = 'string';
    // tableTemplateName.sortOrder = 'asc';
    tableTemplateName.cellTemplate = 'bold';
    tableTemplateName.priority = 9;
    tableTemplateName.allowExporting = true;
    tableTemplates.push(tableTemplateName);

    const tableTemplateCode: TableTemplate = new TableTemplate();
    tableTemplateCode.cssClass = '';
    tableTemplateCode.alignment = 'center';
    tableTemplateCode.caption = 'Código';
    tableTemplateCode.dataField = 'code';
    tableTemplateCode.dataType = 'string';
    tableTemplateCode.priority = 8;
    tableTemplateCode.allowExporting = true;
    tableTemplates.push(tableTemplateCode);

    const tableTemplateWeight: TableTemplate = new TableTemplate();
    tableTemplateWeight.cssClass = '';
    tableTemplateWeight.alignment = 'center';
    tableTemplateWeight.caption = 'Peso';
    tableTemplateWeight.dataField = 'maxWeight';
    tableTemplateWeight.dataType = 'number';
    // tableTemplateWeight.cellTemplate = 'weight';
    tableTemplateWeight.priority = 7;
    tableTemplateWeight.allowExporting = true;
    tableTemplates.push(tableTemplateWeight);

    const tableTemplateDepth: TableTemplate = new TableTemplate();
    tableTemplateDepth.cssClass = '';
    tableTemplateDepth.alignment = 'center';
    tableTemplateDepth.caption = 'Largo';
    tableTemplateDepth.dataField = 'maxDepth';
    tableTemplateDepth.dataType = 'number';
    // tableTemplateDepth.cellTemplate = 'capacity';
    tableTemplateDepth.priority = 6;
    tableTemplateDepth.allowExporting = true;
    tableTemplates.push(tableTemplateDepth);

    const tableTemplateWidth: TableTemplate = new TableTemplate();
    tableTemplateWidth.cssClass = '';
    tableTemplateWidth.alignment = 'center';
    tableTemplateWidth.caption = 'Ancho';
    tableTemplateWidth.dataField = 'maxWidth';
    tableTemplateWidth.dataType = 'string';
    // tableTemplateWidth.cellTemplate = 'capacity';
    tableTemplateWidth.priority = 5;
    tableTemplateWidth.allowExporting = true;
    tableTemplates.push(tableTemplateWidth);

    const tableTemplateHeight: TableTemplate = new TableTemplate();
    tableTemplateHeight.cssClass = '';
    tableTemplateHeight.alignment = 'center';
    tableTemplateHeight.caption = 'Alto';
    tableTemplateHeight.dataField = 'maxHeight';
    tableTemplateHeight.dataType = 'number';
    // tableTemplateHeight.cellTemplate = 'capacity';
    tableTemplateHeight.priority = 4;
    tableTemplateHeight.allowExporting = true;
    tableTemplates.push(tableTemplateHeight);

    const tableTemplateDateCreation: TableTemplate = new TableTemplate();
    tableTemplateDateCreation.cssClass = '';
    tableTemplateDateCreation.caption = 'Fecha de creación';
    tableTemplateDateCreation.dataField = 'dateCreation';
    tableTemplateDateCreation.dataType = 'Date';
    tableTemplateDateCreation.priority = 3;
    // tableTemplateDateCreation.cellTemplate = 'capacity';
    tableTemplateDateCreation.allowExporting = true;
    tableTemplates.push(tableTemplateDateCreation);

    const tableTemplateDateUpdate: TableTemplate = new TableTemplate();
    tableTemplateDateUpdate.cssClass = '';
    tableTemplateDateUpdate.caption = 'Fecha de actualización';
    tableTemplateDateUpdate.dataField = 'dateLastUpdate';
    tableTemplateDateUpdate.dataType = 'Date';
    tableTemplateDateUpdate.priority = 2;
    // tableTemplateDateUpdate.cellTemplate = 'capacity';
    tableTemplateDateUpdate.allowExporting = true;
    tableTemplates.push(tableTemplateDateUpdate);

    const tableTemplateActive: TableTemplate = new TableTemplate();
    tableTemplateActive.cssClass = '';
    tableTemplateActive.caption = 'Activo';
    tableTemplateActive.dataField = 'active';
    tableTemplateActive.dataType = 'text';
    tableTemplateActive.priority = 1;
    tableTemplateActive.allowExporting = true;
    tableTemplates.push(tableTemplateActive);

    return tableTemplates;
  }

  /**
   *  Establece el formato visual del formulario
   *  @param type tipo de formato a establecer
   *  @return FormSegment[]: retorna un arreglo de objetos FormSegment
   */
  public setFormSegment(type: string): FormSegment[] {
    switch (type) {
    case 'simple':
      return this.getFormTemplateViewCapacity();
    default:
      return [];
    }
  }

  public getFormTemplateViewCapacity(): FormSegment[] {
    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = 'Información general';
    formSegmentGeneralInformation.layoutCss = 'grid grid-cols-2 pt-3';
    formSegmentGeneralInformation.templates = [];

    const formTemplateName: FormTemplate = new FormTemplate();
    formTemplateName.field = 'name';
    formTemplateName.name = 'Nombre';
    formTemplateName.placeholder = 'Nombre';
    formTemplateName.type = 'simple-text';
    formTemplateName.isRequired = true;
    formTemplateName.cssClass = 'my-2';
    formTemplateName.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateName);

    const formTemplateCode: FormTemplate = new FormTemplate();
    formTemplateCode.field = 'code';
    formTemplateCode.name = 'Código';
    formTemplateCode.placeholder = 'Código';
    formTemplateCode.type = 'simple-text';
    formTemplateCode.isRequired = true;
    formTemplateCode.cssClass = 'my-2';
    formTemplateCode.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateCode);

    formSegments.push(formSegmentGeneralInformation);

    const formSegmentCapacity: FormSegment = new FormSegment();
    formSegmentCapacity.name = 'Capacidad';
    formSegmentCapacity.layoutCss = "grid grid-cols-2 pt-3";
    formSegmentCapacity.templates = [];

    const formTemplateWeight: FormTemplate = new FormTemplate();
    formTemplateWeight.field = 'maxWeight';
    formTemplateWeight.name = 'Peso maximo';
    formTemplateWeight.placeholder = 'Peso (lb)';
    formTemplateWeight.type = 'simple-number';
    formTemplateWeight.isRequired = true;
    formTemplateWeight.icon = "ri-bar-chart-horizontal-line";
    formTemplateWeight.cssClass = 'my-2';
    formSegmentCapacity.templates.push(formTemplateWeight);

    const formTemplateDepth: FormTemplate = new FormTemplate();
    formTemplateDepth.field = 'maxDepth';
    formTemplateDepth.name = 'Largo maximo';
    formTemplateDepth.placeholder = 'largo (cm)';
    formTemplateDepth.type = 'simple-number';
    formTemplateDepth.isRequired = true;
    formTemplateDepth.cssClass = 'my-2';
    formTemplateDepth.icon = "ri-bar-chart-horizontal-line";
    formSegmentCapacity.templates.push(formTemplateDepth);

    const formTemplateWidth: FormTemplate = new FormTemplate();
    formTemplateWidth.field = 'maxWidth';
    formTemplateWidth.name = 'Ancho maximo';
    formTemplateWidth.placeholder = 'Ancho (cm)';
    formTemplateWidth.type = 'simple-number';
    formTemplateWidth.isRequired = true;
    formTemplateWidth.cssClass = 'my-2';
    formTemplateWidth.icon = 'ri-bar-chart-horizontal-line';
    formSegmentCapacity.templates.push(formTemplateWidth);

    const formTemplateHeight: FormTemplate = new FormTemplate();
    formTemplateHeight.field = 'maxHeight';
    formTemplateHeight.name = 'Alto maximo';
    formTemplateHeight.placeholder = 'Alto (cm)';
    formTemplateHeight.type = 'simple-number';
    formTemplateHeight.isRequired = true;

    formSegmentCapacity.templates.push(formTemplateHeight);

}
