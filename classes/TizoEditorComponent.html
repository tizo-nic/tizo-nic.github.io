import { TizoViewComponent } from './tizo-view.component';

// Forms
import { UntypedFormBuilder, UntypedFormGroup } from '@angular/forms';

// RXJS
import { Observable } from 'rxjs';

// Router
import { Router } from '@angular/router';
import {FormSelectable} from '../support/form-selectable';

// Imports spinner
import {NgxSpinnerService} from "ngx-spinner";

export class TizoEditorComponent<T> extends TizoViewComponent<T> {

  public formGroup: UntypedFormGroup;
  public selectables: { [name: string]: FormSelectable; } | any;
  public suggestions: { [name: string]: (text: Observable<string>) => Observable<readonly any[]> };
  public errorMessage: string | any;
  public isSubmitted: boolean;

  constructor(
    protected readonly _spinner: NgxSpinnerService,
    protected readonly _formBuilder: UntypedFormBuilder,
    protected override readonly _router: Router,
  ) {
    super(_router);
    this.formGroup = this._formBuilder.group({});
    this.selectables = {};
    this.suggestions = {};
    this.isSubmitted = false;
  }

  public override onInit(): void {
    super.onInit();
    setTimeout(() => this.loadData(), 0);
  }
  public actionSave(): void {}

  public actionCancel(): void {

  }

  public actionRemove(data: T): void {}

  public actionUpdate(data: T): void {}

  public override onChangeLanguage() {
    super.onChangeLanguage();
  }
}
