import {TizoEditorComponent} from './tizo-editor.component';
import {MapsAPILoader} from '@agm/core';
import {MapService} from '../../services/map.service';
import {UntypedFormBuilder} from '@angular/forms';
import { Router } from '@angular/router';
import { NgZone } from '@angular/core';
import {NgxSpinnerService} from "ngx-spinner";

declare const google: any;

export class TizoMapEditorComponent<T> extends TizoEditorComponent<T> {

  public zoom: number;
  public latitude: number;
  public longitude: number;

  public styles: any[];

  public drawingManagerOptions: any;
  public drawingManager: any;

  public map: google.maps.Map;
  public polygon: google.maps.Polygon;
  public polygons: google.maps.Polygon[];
  public polygonOptions: google.maps.PolygonOptions;

  constructor(
    protected override readonly _spinner: NgxSpinnerService,
    protected override readonly _formBuilder: UntypedFormBuilder,
    protected readonly _mapsAPILoader: MapsAPILoader,
    protected override readonly _router: Router,
    protected readonly _ngZone: NgZone,
    protected readonly _mapService: MapService,
  ) {
    super(_spinner, _formBuilder, _router);

    this.zoom = 11;


    // Obtain the current location of the user
    this._mapService.getPosition().then((position :any) =>{
      this.latitude = position.lat;
      this.longitude = position.lng;
    });

    this._mapsAPILoader.load().then(() => {
      this.drawingManagerOptions = {
        drawingControl: true,
        drawingControlOptions: {
          drawingModes: ['polygon'],
        },
        polygonOptions: {
          draggable: true,
          editable: true,
          strokeColor: '#13a30a',
          strokeOpacity: 0.8,
          strokeWeight: 3,
          fillColor: '#3fbabd',
          fillOpacity: 0.35
        },
        drawingMode: google.maps.drawing.OverlayType.POLYGON // 'polygon',
      };
      this.placesMap();
    });

    this.polygonOptions = {
      editable: true,
      strokeColor: '#979797',
      strokeOpacity: 0.8,
      strokeWeight: 3,
      fillColor: '#739245', // this.constants.color.primary,
      fillOpacity: 0.35
    };

    this.polygons = [];
  }

  public override onInit(): void {
    this.onChangeLanguage();
  }

  public onMapReady(map: google.maps.Map): void {
    this.map = map;
  }

  public override onChangeLanguage() {
  }

  public override onDestroy(): void {
    super.onDestroy();
  }

  protected initializeMap(): void {
  }

  protected initializeDrawingManager = () => {
    if (this.drawingManager) {
      this.drawingManager.setDrawingMode(null);
      this.drawingManager.setOptions({drawingControl: false});
    }
    this.drawingManager = new google.maps.drawing.DrawingManager(this.drawingManagerOptions);

    this.drawingManager.setMap(this.map);
    google.maps.event.addListener(this.drawingManager, 'overlaycomplete',
      (event: any) => {
        this.polygonOverlayComplete(event);
      });
  };

  protected polygonOverlayComplete(event: any): void {
  }

  protected polygonInsertAt(event: any): void {
  }

  protected polygonSetAt(event: any): void {
  }

  protected polygonRemoveAt(event: any): void {
  }

  protected polygonClick(event: any, polygon: google.maps.Polygon): void {
  }

  protected generatePolygon(paths: any[], center: boolean): google.maps.Polygon {
    const polygon = new google.maps.Polygon(this._mapService.extend(this.polygonOptions, {paths: paths}));

    polygon.setMap(this.map);

    google.maps.event.addListener(polygon.getPath(), 'insert_at', (event: any) => this.polygonInsertAt(event));

    google.maps.event.addListener(polygon.getPath(), 'set_at', (event: any) => this.polygonSetAt(event));

    google.maps.event.addListener(polygon.getPath(), 'remove_at', (event: any) => this.polygonRemoveAt(event));

    google.maps.event.addListener(polygon, 'click', (event: any) => this.polygonClick(event, polygon));

    if (center) this.centerPolygon(paths);

    return polygon;
  }

  protected centerPolygon(paths: any[]): void {
    const latLngBounds = new google.maps.LatLngBounds();

    paths[0].forEach((coordinate: any) => latLngBounds.extend(coordinate));

    const marketCenter = latLngBounds.getCenter();

    this.latitude = marketCenter.lat();
    this.longitude = marketCenter.lng();
    this.zoom = 11;
  }

  public placesMap() {}

}
