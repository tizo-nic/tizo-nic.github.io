import {FormInterface} from "../../interface/form.interface";
import {TableInterface} from "../../interface/table.interface";
import {FormSegment} from "../../support/form-segment";
import {TableTemplate} from "../../support/table-template";
import {FormTemplate} from "../../support/form-template";

export class MeasurementEntity implements  FormInterface, TableInterface{

  public idCountry: number;
  public idMeasure: number;
  public distance: string;
  public longitude: string;
  public weight: string;
  public time: string;
  public dateCreation: Date;
  public dateLastUpdate: Date;


  formSegments: FormSegment[];
  tableTemplates: TableTemplate[];

  public setFormSegment(type: string, fieldShow?: boolean): FormSegment[] {
    switch (type) {
    case 'simple':
      return this.getFormSegmentSimple();
    default:
      return [];
    }
  }

  public getFormSegmentSimple(): FormSegment[] {
    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = 'Información general de las unidades de medida';
    formSegmentGeneralInformation.layoutCss = 'grid grid-cols-2 pt-3';
    formSegmentGeneralInformation.templates = [];


    const formTemplateWeight: FormTemplate = new FormTemplate();
    formTemplateWeight.field = 'distance';
    formTemplateWeight.name = 'Distancia';
    formTemplateWeight.placeholder = 'ej: mi, km, m, cm';
    formTemplateWeight.type = 'simple-text';
    formTemplateWeight.isRequired = true;
    formTemplateWeight.icon = "ri-pin-distance-line";
    formTemplateWeight.cssClass = 'my-2';
    formSegmentGeneralInformation.templates.push(formTemplateWeight);

    const formTemplateDepth: FormTemplate = new FormTemplate();
    formTemplateDepth.field = 'longitude';
    formTemplateDepth.name = 'longitud';
    formTemplateDepth.placeholder = 'ej: cm, in, mm';
    formTemplateDepth.type = 'simple-text';
    formTemplateDepth.isRequired = true;
    formTemplateDepth.cssClass = 'my-2';
    formTemplateDepth.icon = "ri-bar-chart-horizontal-line";
    formSegmentGeneralInformation.templates.push(formTemplateDepth);

    const formTemplateWidth: FormTemplate = new FormTemplate();
    formTemplateWidth.field = 'weight';
    formTemplateWidth.name = 'Peso';
    formTemplateWidth.placeholder = 'ej: kg, lb, g';
    formTemplateWidth.type = 'simple-text';
    formTemplateWidth.isRequired = true;
    formTemplateWidth.cssClass = 'my-2';
    formTemplateWidth.icon = 'ri-bar-chart-horizontal-line';
    formSegmentGeneralInformation.templates.push(formTemplateWidth);

    const formTemplateHeight: FormTemplate = new FormTemplate();
    formTemplateHeight.field = 'time';
    formTemplateHeight.name = 'Tiempo';
    formTemplateHeight.placeholder = 'ej: h, min, s';
    formTemplateHeight.type = 'simple-text';
    formTemplateHeight.isRequired = true;
    formTemplateHeight.cssClass = 'my-2';
    formTemplateHeight.icon = 'ri-time-line';
    formSegmentGeneralInformation.templates.push(formTemplateHeight);

    formSegments.push(formSegmentGeneralInformation);

    return formSegments;
  }



  public setTableTemplate(type: string): TableTemplate[] {
    switch (type) {
    case 'simple':
      return this.getTableTemplateViewMeasurement();
    default:
      return [];
    }
  }

  /**
   * Obtiene el formato visual de la tabla
   * @return {TableTemplate}
   */
  public getTableTemplateViewMeasurement(): TableTemplate[] {
    const tableTemplates: TableTemplate[] = [];

    const tableTemplateOptions: TableTemplate = new TableTemplate();
    tableTemplateOptions.alignment = 'center';
    tableTemplateOptions.cssClass = '';
    tableTemplateOptions.caption = 'Opciones';
    tableTemplateOptions.cellTemplate = 'options';
    tableTemplateOptions.priority = 8;
    tableTemplateOptions.allowExporting = false;
    tableTemplates.push(tableTemplateOptions);

    const tableTemplateDistance: TableTemplate = new TableTemplate();
    tableTemplateDistance.cssClass = 'dx-column-middle';
    tableTemplateDistance.alignment = 'center';
    tableTemplateDistance.caption = 'Distancia';
    tableTemplateDistance.dataField = 'distance';
    tableTemplateDistance.dataType = 'string';
    //tableTemplateDistance.cellTemplate = 'measurement';
    tableTemplateDistance.priority = 7;
    tableTemplateDistance.allowExporting = true;
    tableTemplates.push(tableTemplateDistance);

    const tableTemplateLength: TableTemplate = new TableTemplate();
    tableTemplateLength.cssClass = '';
    tableTemplateLength.alignment = 'center';
    tableTemplateLength.caption = 'Longitud';
    tableTemplateLength.dataField = 'longitude';
    tableTemplateLength.dataType = 'string';
    //tableTemplateLength.cellTemplate = 'measurement';
    tableTemplateLength.priority = 6;
    tableTemplateLength.allowExporting = true;
    tableTemplates.push(tableTemplateLength);

    const tableTemplateWeight: TableTemplate = new TableTemplate();
    tableTemplateWeight.cssClass = '';
    tableTemplateWeight.alignment = 'center';
    tableTemplateWeight.caption = 'Peso';
    tableTemplateWeight.dataField = 'weight';
    tableTemplateWeight.dataType = 'string';
    //tableTemplateWeight.cellTemplate = 'measurement';
    tableTemplateWeight.priority = 5;
    tableTemplateWeight.allowExporting = true;
    tableTemplates.push(tableTemplateWeight);

    const tableTemplateTime: TableTemplate = new TableTemplate();
    tableTemplateTime.cssClass = '';
    tableTemplateTime.alignment = 'center';
    tableTemplateTime.caption = 'Tiempo';
    tableTemplateTime.dataField = 'time';
    tableTemplateTime.dataType = 'Date';
    //tableTemplateTime.cellTemplate = 'measurement';
    tableTemplateTime.priority = 3;
    tableTemplateTime.allowExporting = true;
    tableTemplates.push(tableTemplateTime);


    const tableTemplateDateCreation: TableTemplate = new TableTemplate();
    tableTemplateDateCreation.cssClass = '';
    tableTemplateDateCreation.caption = 'Fecha de creación';
    tableTemplateDateCreation.dataField = 'dateCreation';
    tableTemplateDateCreation.dataType = 'date';
    tableTemplateDateCreation.priority = 2;
    //tableTemplateDateCreation.cellTemplate = 'measurement';
    tableTemplateDateCreation.allowExporting = true;
    tableTemplates.push(tableTemplateDateCreation);

    const tableTemplateDateUpdate: TableTemplate = new TableTemplate();
    tableTemplateDateUpdate.cssClass = '';
    tableTemplateDateUpdate.caption = 'Fecha de actualización';
    tableTemplateDateUpdate.dataField = 'dateLastUpdate';
    tableTemplateDateUpdate.dataType = 'date';
    tableTemplateDateUpdate.priority = 1;
    //tableTemplateDateUpdate.cellTemplate = 'capacity';
    tableTemplateDateUpdate.allowExporting = true;
    tableTemplates.push(tableTemplateDateUpdate);

    return tableTemplates;
  }




}
