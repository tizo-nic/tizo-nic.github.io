import { FormSegment } from '../../support/form-segment';

// Interface
import { FormInterface } from '../../interface/form.interface';
import { TableInterface } from '../../interface/table.interface';

// Templates for the component
import { FormTemplate } from '../../support/form-template';
import { TableTemplate } from '../../support/table-template';

export class CountryEntity implements FormInterface, TableInterface {

  public idCountry: number;
  public name: string;
  public isoCode: string;
  public latitude: number;
  public longitude: number;
  public formatPhone: string;
  public iconPath: string;
  public dateCreation: Date;
  public dateUpdate: Date;
  public status: number;


  formSegments: FormSegment[];
  tableTemplates: TableTemplate[];

  public setTableTemplate(type: string): TableTemplate[] {
    switch (type) {
    case 'simple':
      return this.getTableTemplateSimple();
    default:
      return [];
    }
  }

    const tableTemplates: TableTemplate[] = [];

    const tableTemplateOptions: TableTemplate = new TableTemplate();
    tableTemplateOptions.alignment = 'center';
    tableTemplateOptions.caption = 'Opciones';
    tableTemplateOptions.cellTemplate = 'options';
    tableTemplateOptions.priority = 5;
    tableTemplateOptions.allowExporting = false;
    tableTemplateOptions.cssClass = '';
    tableTemplates.push(tableTemplateOptions);

    const tableTemplateName: TableTemplate = new TableTemplate();
    tableTemplateName.alignment = 'center';
    tableTemplateName.dataField = 'name';
    tableTemplateName.caption = 'País';
    tableTemplateName.dataType = 'string';
    tableTemplateName.cssClass = '';
    tableTemplateName.cellTemplate = 'bold';
    tableTemplateName.priority = 4;
    tableTemplateName.allowExporting = true;
    tableTemplates.push(tableTemplateName);

    const tableTemplateIso: TableTemplate = new TableTemplate();
    tableTemplateIso.alignment = 'center';
    tableTemplateIso.caption = 'Código ISO';
    tableTemplateIso.dataField = 'code';
    tableTemplateIso.dataType = 'string';
    tableTemplateIso.cssClass = '';
    tableTemplateIso.priority = 3;
    tableTemplateIso.allowExporting = true;
    tableTemplates.push(tableTemplateIso);

    const tableTemplatePhone: TableTemplate = new TableTemplate();
    tableTemplatePhone.alignment = 'center';
    tableTemplatePhone.caption = 'Formato de teléfono';
    tableTemplatePhone.dataField = 'formatPhone';
    tableTemplatePhone.dataType = 'string';

    tableTemplates.push(tableTemplatePhone);

    const tableTemplateDateCreation: TableTemplate = new TableTemplate();
    tableTemplateDateCreation.alignment = 'center';
    tableTemplateDateCreation.caption = 'Fecha de creación';
    tableTemplateDateCreation.dataField = 'dateCreation';
    tableTemplateDateCreation.dataType = 'date';
    tableTemplateDateCreation.cssClass = '';
    tableTemplateDateCreation.priority = 2;
    tableTemplateDateCreation.allowExporting = true;
    tableTemplates.push(tableTemplateDateCreation);

    const tableTemplateDateUpdate: TableTemplate = new TableTemplate();
    tableTemplateDateUpdate.alignment = 'center';
    tableTemplateDateUpdate.caption = 'Fecha de actualización';
    tableTemplateDateUpdate.dataField = 'dateUpdate';
    tableTemplateDateUpdate.dataType = 'date';
    tableTemplateDateUpdate.cssClass = '';
    tableTemplateDateUpdate.priority = 1;
    tableTemplateDateUpdate.allowExporting = true;
    tableTemplates.push(tableTemplateDateUpdate);

    return tableTemplates;
  }

  public setFormSegment(type: string): FormSegment[] {
    switch (type) {
    case 'simple':
      return this.getFormSegmentSimple();
    default:
      return [];
    }
  }

    const formSegments: FormSegment[] = [];

    formSegmentGeneralInformation.name = 'Información general';
    formSegmentGeneralInformation.layoutCss = 'grid grid-cols-2 pt-3';

    const formTemplateName: FormTemplate = new FormTemplate();
    formTemplateName.field = 'name';
    formTemplateName.name = 'País';
    formTemplateName.placeholder = 'ej: Nicaragua';
    formTemplateName.type = 'simple-text';

    formSegmentGeneralInformation.templates.push(formTemplateIso);

    const formTemplateProfile: FormTemplate = new FormTemplate();
    formTemplateProfile.field = 'idsCurrencies';
    formTemplateProfile.name = 'text_currencies';
    formTemplateProfile.placeholder = 'text_currencies';
    formTemplateProfile.type = 'multi-selector';
    formTemplateProfile.isRequired = true;
    formTemplateProfile.cssClass = 'col-md-4';

    formSegmentGeneralInformation.templates.push(formTemplateProfile);

    const formTemplateFormatPhone: FormTemplate = new FormTemplate();
    formTemplateFormatPhone.field = 'formatPhone';

    formTemplateFormatPhone.placeholder = '+(###) ####-####';
    formTemplateFormatPhone.type = 'simple-text';
    formTemplateFormatPhone.isRequired = true;
    formTemplateFormatPhone.cssClass = '';
    formTemplateFormatPhone.icon = 'ri-smartphone-line';
    formTemplateFormatPhone.isHasTooltip = true;

    formSegments.push(formSegmentGeneralInformation);

    const formSegmentLocation: FormSegment = new FormSegment();

    formSegmentLocation.templates = [];

    const formTemplateLatitude: FormTemplate = new FormTemplate();

    formTemplateLatitude.name = 'Latitud';

    formTemplateLatitude.type = 'simple-number';
    formTemplateLatitude.isRequired = true;
    formTemplateLatitude.cssClass = 'my-2';
    formTemplateLatitude.icon = 'ri-bar-chart-horizontal-line';
    formSegmentLocation.templates.push(formTemplateLatitude);

    const formTemplateLongitude: FormTemplate = new FormTemplate();
    formTemplateLongitude.field = 'longitude';
    formTemplateLongitude.name = 'Longitud';

    formTemplateLongitude.cssClass = 'my-2';
    formTemplateLongitude.icon = 'ri-bar-chart-horizontal-line';
    formSegmentLocation.templates.push(formTemplateLongitude);

  }

}
