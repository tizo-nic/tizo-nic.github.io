import { TableInterface } from '../../interface/table.interface';

// Templates for the component
import { TableTemplate } from '../../support/table-template';

export class DocumentListEntity implements TableInterface {
  public tableTemplates: TableTemplate[];

  public idDocumentList: number;
  public idCountry: number;
  public relation: number;

  public isTemplate: boolean;

  public dateCreation: Date;
  public dateUpdate: Date;
  public status: number;

  public setTableTemplate(type: string): TableTemplate[] {
    switch (type) {
    case 'simple':
      return this.getTableTemplateSimple();
    default:
      return [];
    }
  }

  public getTableTemplateSimple(): TableTemplate[] {

    let tableTemplateOptions: TableTemplate;
    tableTemplateOptions = new TableTemplate();
    tableTemplateOptions.width = 140;
    tableTemplateOptions.alignment = 'center';
    tableTemplateOptions.cssClass = 'dx-column-middle';
    tableTemplateOptions.caption = 'text_options';
    tableTemplateOptions.cellTemplate = 'options';
    tableTemplateOptions.priority = 7;
    tableTemplateOptions.allowExporting = false;
    this.tableTemplates.push(tableTemplateOptions);

    return this.tableTemplates;
  }
}
