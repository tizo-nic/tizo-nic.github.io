import { FormSegment } from '../../support/form-segment';

// Interface
import { FormInterface } from '../../interface/form.interface';
import { TableInterface } from '../../interface/table.interface';

// Templates for the component
import { FormTemplate } from '../../support/form-template';
import { TableTemplate } from '../../support/table-template';

// Entity
import { CountryEntity } from './country.entity';
import { AreaEntity } from './area.entity';

export class BankEntity implements FormInterface, TableInterface {

  public tableTemplates: TableTemplate[] = [];
  public formSegments: FormSegment[];

  public idBank: number;
  public idCountry: number;
  public idArea: number;

  public area: AreaEntity;
  public areaName: string;

  public country: CountryEntity;
  public countryName: string;

  public name: string;
  public url: string;
  public address: string;
  public phone: string;

  public dateCreation: Date;
  public dateUpdate: Date;
  public status: number;

  public setTableTemplate(type: string): TableTemplate[] {
    switch (type) {
    case 'simple':
      return this.getTableTemplateSimple();
    default:
      return [];
    }
  }

    tableTemplateOptions.cssClass = 'dx-column-middle';

    tableTemplateOptions.priority = 7;
    tableTemplateOptions.allowExporting = false;
    this.tableTemplates.push(tableTemplateOptions);

    const tableTemplateName: TableTemplate = new TableTemplate();
    tableTemplateName.cssClass = 'dx-column-middle';
    tableTemplateName.caption = 'text_name';
    tableTemplateName.dataField = 'name';
    tableTemplateName.dataType = 'string';
    tableTemplateName.sortOrder = 'asc';
    tableTemplateName.priority = 6;
    tableTemplateName.allowExporting = true;
    this.tableTemplates.push(tableTemplateName);

    const tableTemplateCountry: TableTemplate = new TableTemplate();
    tableTemplateCountry.cssClass = 'dx-column-middle';
    tableTemplateCountry.caption = 'text_country';
    tableTemplateCountry.dataField = 'countryName';
    tableTemplateCountry.dataType = 'string';
    tableTemplateCountry.priority = 5;
    tableTemplateCountry.groupIndex = 0;
    tableTemplateCountry.allowExporting = true;
    this.tableTemplates.push(tableTemplateCountry);

    // const tableTemplateNameArea: TableTemplate = new TableTemplate();

    const tableTemplateUrl: TableTemplate = new TableTemplate();
    tableTemplateUrl.cssClass = 'dx-column-middle';
    tableTemplateUrl.caption = 'Url';
    tableTemplateUrl.dataField = 'url';
    tableTemplateUrl.dataType = 'string';
    tableTemplateUrl.sortOrder = 'asc';
    tableTemplateUrl.priority = 3;
    tableTemplateUrl.allowExporting = true;
    this.tableTemplates.push(tableTemplateUrl);

    const tableTemplateAddress: TableTemplate = new TableTemplate();
    tableTemplateAddress.cssClass = 'dx-column-middle';
    tableTemplateAddress.caption = 'text_address';
    tableTemplateAddress.dataField = 'address';
    tableTemplateAddress.dataType = 'string';
    tableTemplateAddress.sortOrder = 'asc';
    tableTemplateAddress.priority = 2;
    tableTemplateAddress.allowExporting = true;
    this.tableTemplates.push(tableTemplateAddress);

    const tableTemplatePhone: TableTemplate = new TableTemplate();
    tableTemplatePhone.cssClass = 'dx-column-middle';
    tableTemplatePhone.caption = 'text_phone';
    tableTemplatePhone.dataField = 'phone';
    tableTemplatePhone.dataType = 'string';
    tableTemplatePhone.sortOrder = 'asc';
    tableTemplatePhone.priority = 1;
    tableTemplatePhone.allowExporting = true;
    this.tableTemplates.push(tableTemplatePhone);

  }

  public setFormSegment(type: string): FormSegment[] {
    switch (type) {
    case 'simple':
      return this.getFormSegmentSimple();
    default:
      return [];
    }
  }

  public getFormSegmentSimple(): FormSegment[] {
    const formSegments: FormSegment[] = [];

    const formSegmentGeneralInformation: FormSegment = new FormSegment();
    formSegmentGeneralInformation.name = 'text_general_information';
    formSegmentGeneralInformation.templates = [];

    const formTemplateName: FormTemplate = new FormTemplate();
    formTemplateName.field = 'name';
    formTemplateName.name = 'text_name';
    formTemplateName.placeholder = 'text_name';
    formTemplateName.type = 'simple-text';
    formTemplateName.isRequired = true;
    formTemplateName.cssClass = 'col-md-6';
    formSegmentGeneralInformation.templates.push(formTemplateName);

    const formTemplatePhone: FormTemplate = new FormTemplate();
    formTemplatePhone.field = 'phone';
    formTemplatePhone.name = 'text_phone';
    formTemplatePhone.placeholder = 'text_phone';
    formTemplatePhone.type = 'simple-phone';
    formTemplatePhone.isRequired = false;
    formTemplatePhone.isPhone = true;
    formTemplatePhone.cssClass = 'col-md-6';
    formSegmentGeneralInformation.templates.push(formTemplatePhone);

    const formTemplateUrl: FormTemplate = new FormTemplate();
    formTemplateUrl.field = 'url';
    formTemplateUrl.name = 'Url';
    formTemplateUrl.placeholder = 'Url';
    formTemplateUrl.type = 'simple-text';
    formTemplateUrl.isRequired = false;
    formTemplateUrl.cssClass = 'col-md-12';
    formSegmentGeneralInformation.templates.push(formTemplateUrl);

    formSegments.push(formSegmentGeneralInformation);

    const formSegmentLocation: FormSegment = new FormSegment();
    formSegmentLocation.name = 'text_location';
    formSegmentLocation.templates = [];

    const formTemplateCountry: FormTemplate = new FormTemplate();
    formTemplateCountry.field = 'idCountry';
    formTemplateCountry.name = 'text_country';
    formTemplateCountry.placeholder = 'text_country';
    formTemplateCountry.type = 'simple-selector';
    formTemplateCountry.isRequired = true;
    formTemplateCountry.cssClass = 'col-md-12';
    formSegmentLocation.templates.push(formTemplateCountry);

    const formTemplateAddress: FormTemplate = new FormTemplate();
    formTemplateAddress.field = 'address';
    formTemplateAddress.name = 'text_address';
    formTemplateAddress.placeholder = 'text_address';
    formTemplateAddress.type = 'simple-text';
    formTemplateAddress.isRequired = false;
    formTemplateAddress.cssClass = 'col-md-12';
    formSegmentLocation.templates.push(formTemplateAddress);

    formSegments.push(formSegmentLocation);
    return formSegments;
  }

}
